// UnifyAddress 将以太坊地址统一为标准格式（校验和格式）
// 参数：
//   - address: 原始地址字符串
// 返回值：
//   - string: 统一格式化后的地址
//   - error: 处理过程中的错误信息
func UnifyAddress(address string) (string, error) {
	// 检查地址长度是否小于等于2（以太坊地址长度为42，包含0x前缀）
	// 同时使用go-ethereum库验证地址是否是合法的以太坊十六进制地址
	if len(address) <= 2 || !common.IsHexAddress(address) {
		return "", errors.New("user address is illegal") // 地址非法，返回错误
	}

	// 将地址转换为EIP-55标准的校验和格式（混合大小写校验和）
	// EIP-55通过对地址进行哈希运算，将特定位置的字母大写来实现校验
	addr, err := eip.ToCheckSumAddress(address)
	if err != nil {
		// 地址转换失败，包装原始错误并返回
		return "", errors.Wrap(err, "invalid address")
	}

	// 验证转换后的校验和地址是否通过项目自定义的地址验证规则
	// utils.ToValidateAddress可能实现了额外的地址验证逻辑
	if addr != utils.ToValidateAddress(addr) {
		// 地址统一化验证失败，返回错误
		return "", errors.Wrap(err, "failed on unify address")
	}

	// 地址验证通过，返回统一格式化后的校验和地址
	return addr, nil
}

// VerifySig 验证以太坊签名是否有效
// 参数:
//   - addr: 预期签名者的以太坊地址
//   - sigHex: 十六进制编码的签名数据
//   - digest: 被签名的消息哈希（摘要）
// 返回值:
//   - bool: 签名是否有效
func VerifySig(addr, sigHex string, digest []byte) bool {
	// 将十六进制签名字符串解码为字节数组
	signature := hexutil.Decode(sigHex)
	// 检查签名长度是否为65字节（标准以太坊签名长度：r(32) + s(32) + v(1)）
	if len(signature) != 65 {
		return false
	}
	// 检查恢复ID（v值）是否为27或28（以太坊签名的标准v值）
	if signature[64] != 27 && signature[64] != 28 {
		return false
	}
	// 将v值从27/28转换为0/1（EIP-155之前的标准）
	signature[64] -= 27
	
	// 从签名和摘要中恢复公钥
	publicKeyBytes, err := crypto.Ecrecover(digest, signature)
	if err != nil {
		return false
	}
	
	// 将公钥字节解析为ECDSA公钥结构体
	publicKeyECDSA, err := crypto.UnmarshalPubkey(publicKeyBytes)
	if err != nil {
		return false
	}
	
	// 从公钥派生以太坊地址并转换为十六进制字符串
	address := crypto.PubkeyToAddress(*publicKeyECDSA).Hex()
	// 比较派生地址与预期地址（不区分大小写）
	if !strings.EqualFold(addr, address) {
		return false
	}

	// 移除签名中的恢复ID（最后一个字节），用于后续验证
	signatureNoRecoverID := signature[:len(signature)-1]
	// 使用公钥验证签名的有效性
	return crypto.VerifySignature(publicKeyBytes, digest, signatureNoRecoverID)
}

// AesEncryptOFB 使用AES-OFB模式加密数据
// OFB（输出反馈模式）将块密码转换为流密码
// 参数:
//   - data: 待加密的原始数据
//   - key: 加密密钥（16/24/32字节对应AES-128/192/256）
// 返回值:
//   - []byte: 加密后的数据（包含IV向量）
//   - error: 加密过程中的错误
func AesEncryptOFB(data []byte, key []byte) ([]byte, error) {
	// 对数据进行PKCS7填充，使其长度为AES块大小（16字节）的倍数
	data = PKCS7Padding(data, aes.BlockSize)
	// 创建AES密码块实例（忽略错误检查，实际应用中应检查err）
	block, _ := aes.NewCipher([]byte(key))
	// 创建输出缓冲区：IV向量 + 加密数据
	out := make([]byte, aes.BlockSize+len(data))
	// 将缓冲区前16字节用作IV（初始化向量）
	iv := out[:aes.BlockSize]
	// 从加密随机源读取IV值
	if _, err := io.ReadFull(rand.Reader, iv); err != nil {
		return nil, err
	}

	// 创建OFB模式的加密流
	stream := cipher.NewOFB(block, iv)
	// 将数据加密到输出缓冲区的后半部分（跳过IV位置）
	stream.XORKeyStream(out[aes.BlockSize:], data)
	return out, nil
}

// AesDecryptOFB 使用AES-OFB模式解密数据
// 参数:
//   - data: 待解密的数据（包含IV向量）
//   - key: 解密密钥（必须与加密时使用的密钥相同）
// 返回值:
//   - []byte: 解密后的原始数据
//   - error: 解密过程中的错误
func AesDecryptOFB(data []byte, key []byte) ([]byte, error) {
	// 创建AES密码块实例
	block, _ := aes.NewCipher([]byte(key))
	// 从输入数据中提取IV向量（前16字节）
	iv := data[:aes.BlockSize]
	// 剩余部分为实际的密文数据
	data = data[aes.BlockSize:]
	// 检查密文长度是否为块大小的倍数
	if len(data)%aes.BlockSize != 0 {
		return nil, fmt.Errorf("data is not a multiple of the block size")
	}

	// 创建输出缓冲区
	out := make([]byte, len(data))
	// 创建OFB模式的解密流（OFB模式下加密和解密操作相同）
	mode := cipher.NewOFB(block, iv)
	// 解密密文数据
	mode.XORKeyStream(out, data)

	// 移除PKCS7填充
	out = PKCS7UnPadding(out)
	return out, nil
}

// PKCS7Padding 实现PKCS#7填充算法
// 在加密前将数据填充到块大小的整数倍
// 填充值为填充字节的数量，每个填充字节都设为该数量
// 参数:
//   - ciphertext: 原始数据
//   - blocksize: 块大小（AES为16字节）
// 返回值:
//   - []byte: 填充后的数据
func PKCS7Padding(ciphertext []byte, blocksize int) []byte {
	// 计算需要填充的字节数
	padding := blocksize - len(ciphertext)%blocksize
	// 创建填充字节切片（每个字节值等于填充长度）
	padtext := bytes.Repeat([]byte{byte(padding)}, padding)
	// 将填充追加到原始数据后
	return append(ciphertext, padtext...)
}

// PKCS7UnPadding 移除PKCS#7填充
// 根据最后一个字节的值确定填充长度并移除填充
// 参数:
//   - origData: 解密后的数据（包含填充）
// 返回值:
//   - []byte: 移除填充后的原始数据
func PKCS7UnPadding(origData []byte) []byte {
	// 获取数据长度
	length := len(origData)
	// 获取最后一个字节，该字节的值表示填充的长度
	unpadding := int(origData[length-1])
	// 返回移除填充后的数据
	return origData[:(length - unpadding)]
}

var (
	validatorM map[string]validator.Func  // 存储自定义验证器函数的映射表，键为验证器名称，值为验证函数
	patternM   map[string]string          // 存储正则表达式模式的映射表，键为验证器名称，值为正则表达式
)

func init() {
	// 初始化自定义验证器映射
	validatorM = map[string]validator.Func{
		"symbol":  rightSymbol,      // 代币符号验证器
		"address": regexpValidator,  // 地址格式验证器（使用正则表达式）
	}
	// 初始化正则表达式模式映射
	patternM = map[string]string{
		"address": `^0x[a-fA-F0-9]{40}$`, // 以太坊地址格式：0x开头，后跟40个十六进制字符
	}
}

var (
	// rightSymbol 验证代币符号是否有效的自定义验证器
	// 规则：代币符号必须是字符串，且长度小于10
	rightSymbol validator.Func = func(fl validator.FieldLevel) bool {
		symbol, ok := fl.Field().Interface().(string)  // 获取字段值并断言为字符串类型
		if ok {
			return len(symbol) < 10  // 长度必须小于10（例如："ETH", "USDT"等符合要求）
		}
		return false
	}

	// regexpValidator 通用的正则表达式验证器
	// 根据验证标签（fl.GetTag()）从patternM中获取对应的正则表达式进行匹配
	regexpValidator validator.Func = func(fl validator.FieldLevel) bool {
		key, _ := fl.Field().Interface().(string)  // 获取字段值（待验证的字符串）
		pattern, ok := patternM[fl.GetTag()]       // 根据标签获取对应的正则表达式
		if ok {
			match, _ := regexp.MatchString(pattern, key)  // 执行正则匹配
			return match
		}
		return false
	}
)

// ToValidateAddress 将以太坊地址转换为校验和格式（EIP-55标准）
// 输入参数：address - 原始的以太坊地址字符串（可能大小写混合）
// 返回值：符合EIP-55标准的校验和地址
func ToValidateAddress(address string) string {
	// 将原始地址转换为小写，便于处理
	addrLowerStr := strings.ToLower(address)
	// 如果地址以"0x"开头，则移除前缀，方便后续处理
	if strings.HasPrefix(addrLowerStr, "0x") {
		addrLowerStr = addrLowerStr[2:]  // 移除前缀后的纯地址部分
		address = address[2:]            // 原始地址也移除前缀，保持一致性
	}
	
	// 创建变量存储处理后的地址字节
	var binaryStr string
	addrBytes := []byte(addrLowerStr)  // 将小写地址转换为字节数组

	// 计算小写地址字符串的Keccak256哈希值
	// 注意：这里是对字符串本身进行哈希，而不是对解码后的字节进行哈希
	hash256 := common.Keccak256Hash([]byte(addrLowerStr))

	// 遍历地址字符串的每个字符
	for i, e := range addrLowerStr {
		// 如果是数字（0-9），则跳过（数字在EIP-55中不进行大小写转换）
		if e >= '0' && e <= '9' {
			continue
		} else {
			// 对于字母字符（a-f），根据哈希值的对应位决定是否转换为大写
			// 将哈希值的第i/2个字节转换为8位二进制字符串，并确保填充到8位（fmt会自动处理）
			binaryStr = fmt.Sprintf("%08b", hash256[i/2])
			
			// 检查哈希值的特定位是否为1
			// i%2用于确定是使用高4位还是低4位（每个哈希字节对应两个地址字符）
			// 4*(i%2)计算偏移量：当i为偶数时使用高4位（索引0），i为奇数时使用低4位（索引4）
			if binaryStr[4*(i%2)] == '1' {
				// 如果哈希位为1，将该字母转换为大写
				// 小写字母和大写字母的ASCII码相差32
				addrBytes[i] -= 32  // 例如：'a'(97) -> 'A'(65)，减去32
			}
		}
	}

	// 添加"0x"前缀并返回处理后的地址
	return "0x" + string(addrBytes)
}

算法细节：

每个地址字符对应哈希值的4位（半字节）

哈希位为1时对应字母大写，为0时保持小写

这个地址校验和实现与标准的EIP-55兼容，确保了地址的唯一性和可验证性，是以太坊生态中的重要安全特性。

// UnmarshalConfig 解析配置文件
// 参数：
//   - configFilePath: 配置文件的完整路径
// 返回值：
//   - *Config: 解析后的配置对象
//   - error: 解析过程中的错误
func UnmarshalConfig(configFilePath string) (*Config, error) {
	// 设置Viper的配置文件路径
	viper.SetConfigFile(configFilePath)
	// 指定配置文件类型为TOML（Tom's Obvious, Minimal Language）
	viper.SetConfigType("toml")
	// 启用自动环境变量覆盖（允许用环境变量覆盖配置文件中的值）
	viper.AutomaticEnv()
	// 设置环境变量前缀为"CNFT"，即所有环境变量必须以CNFT_开头
	viper.SetEnvPrefix("CNFT")
	// 创建字符串替换器，将配置键中的"."替换为"_"
	// 例如：配置项"db.host"可以通过环境变量"CNFT_DB_HOST"覆盖
	replacer := strings.NewReplacer(".", "_")
	viper.SetEnvKeyReplacer(replacer)

	// 读取配置文件内容
	if err := viper.ReadInConfig(); err != nil {
		return nil, err
	}
	
	// 获取默认配置作为基础
	config, err := DefaultConfig()
	if err != nil {
		return nil, err
	}

	// 将读取到的配置解析到config结构体中
	// Viper会自动处理环境变量覆盖和文件内容的合并
	if err := viper.Unmarshal(config); err != nil {
		return nil, err
	}
	
	return config, nil
}

// DefaultConfig 返回默认配置
// 目前返回空配置，可根据需要添加默认值
func DefaultConfig() (*Config, error) {
	return &Config{}, nil
}

// 1. 获取临时上传凭证
func (s *COSService) GetUploadCredential(req *types.COSTokenRequest) (*types.COSTokenResponse, error) {
	// 生成唯一的文件路径
	key := s.generateObjectKey(req.FileType, req.FileName)
	
	// 计算过期时间（2小时）
	expireTime := time.Now().Add(2 * time.Hour)
	
	// STS临时凭证请求
	stsReq := &sts.CredentialOptions{
		SecretID:  s.secretID,
		SecretKey: s.secretKey,
		DurationSeconds: 7200, // 2小时
		Region: s.region,
		Policy: &sts.CredentialPolicy{
			Statement: []sts.CredentialPolicyStatement{
				{
					Action: []string{
						"name/cos:PutObject",    // 允许上传
						"name/cos:PostObject",   // 允许表单上传
					},
					Effect: "allow",
					Resource: []string{
						fmt.Sprintf("qcs::cos:%s:uid/%s:%s/%s",
							s.region, s.appID, s.bucket, key),
					},
				},
			},
		},
	}
	
	// 获取临时凭证
	res, err := sts.NewCredential(stsReq)
	if err != nil {
		return nil, err
	}
	
	// 构建响应
	return &types.COSTokenResponse{
		Credentials: &types.COSCredentials{
			TmpSecretID:  res.Credentials.TmpSecretID,
			TmpSecretKey: res.Credentials.TmpSecretKey,
			SessionToken: res.Credentials.SessionToken,
		},
		RequestID:   res.RequestID,
		Expiration:  expireTime.Format(time.RFC3339),
		StartTime:   time.Now().Format(time.RFC3339),
		ExpiredTime: expireTime.Unix(),
		Bucket:      s.bucket,
		Region:      s.region,
		AllowPrefix: key,
		AllowActions: []string{
			"cos:PutObject",
			"cos:PostObject",
		},
		UploadURL: fmt.Sprintf("https://%s.cos.%s.myqcloud.com", s.bucket, s.region),
		Key:       key,
	}, nil
}

// 生成对象键（文件路径）
func (s *COSService) generateObjectKey(fileType, fileName string) string {
	// 生成唯一ID（基于时间戳和文件名）
	hash := md5.Sum([]byte(fmt.Sprintf("%d-%s", time.Now().UnixNano(), fileName)))
	uniqueID := hex.EncodeToString(hash[:8]) // 取前8位
	
	// 获取文件扩展名
	ext := filepath.Ext(fileName)
	
	// 根据文件类型组织目录结构
	var dir string
	switch fileType {
	case "image":
		dir = "nft/images"
	case "video":
		dir = "nft/videos"
	case "audio":
		dir = "nft/audios"
	case "document":
		dir = "nft/documents"
	default:
		dir = "nft/others"
	}
	
	// 按日期分目录
	dateDir := time.Now().Format("2006/01/02")
	
	// 最终路径: nft/images/2024/01/15/abc12345.png
	return fmt.Sprintf("%s/%s/%s%s", dir, dateDir, uniqueID, ext)
}

🏗️ 函数选项模式（Functional Options Pattern）
这是Go语言中一种非常流行的设计模式，用于灵活地初始化结构体。代码中通过以下元素实现：

CtxOption 类型：定义为一个函数类型，它接收一个*CtxConfig指针。

go
type CtxOption func(conf *CtxConfig)
选项函数（如WithDB）：每个选项函数返回一个CtxOption，这个闭包会修改传入的CtxConfig。

go
func WithDB(db *gorm.DB) CtxOption {
    return func(conf *CtxConfig) {
        conf.db = db
    }
}
构造函数 NewServerCtx：接收可变数量的CtxOption，依次应用它们来配置CtxConfig，最后构建出ServerCtx。

go
func NewServerCtx(options ...CtxOption) *ServerCtx {
    c := &CtxConfig{}
    for _, opt := range options {
        opt(c) // 应用每个选项函数，填充CtxConfig
    }
    return &ServerCtx{
        DB:      c.db,
        KvStore: c.KvStore,
        Dao:     c.dao,
    }
}
💡 为什么使用这种模式？
这种设计带来了几个关键优势，在构建大型应用时尤其有用：

极大的灵活性：可以按需初始化任何依赖组合。例如，测试时可以只传入KvStore而不用真实的数据库。

清晰的扩展性：未来如果需要增加新的依赖（比如一个邮件服务客户端），只需添加一个新的WithEmailClient选项函数，完全无需修改现有的NewServerCtx调用代码。

避免巨型构造函数：避免了像NewServerCtx(db, kv, dao, evm, imageMgr, ...)这样参数列表冗长、难以维护的构造函数。

参数顺序无关：调用时，各个选项的顺序是自由的，提高了代码的可读性。

构建Redis/KV配置：遍历配置文件中的Redis节点信息，将其转换为 go-zero 框架所需的 kv.KvConf 格式。

初始化数据库连接：调用 gdb.NewDB(&c.DB) 建立与数据库的连接，得到 *gorm.DB 实例 db。

初始化多链节点服务：遍历配置中所有支持的链（c.ChainSupported），为每条链创建一个 nftchainservice.Service 实例。这个服务封装了与特定区块链节点交互的逻辑（如获取NFT元数据、监听事件等），并注入了解析元数据所需的标签配置。所有服务按链ID存储在 nodeSrvs map中。

初始化数据访问层：调用 dao.New(context.Background(), db, store) 创建DAO实例，它封装了所有数据库操作。

组合服务上下文：调用之前定义的 NewServerCtx 函数（使用函数选项模式），将 db、store、dao 等核心依赖注入到 ServerCtx 结构体中。

附加其他配置：将原始配置对象 c 和创建好的多链服务map nodeSrvs 也附加到 ServerCtx 上，形成一个包含所有运行时依赖的完整上下文对象。

🧩 ServerCtx 结构体的角色
ServerCtx 作为最终产出的“服务容器”，聚合了所有下游组件可能用到的共享资源：

C：原始配置，供各模块按需读取。

DB 与 Dao：数据持久化层的入口。

KvStore：缓存和队列操作的客户端。

NodeSrvs：一个以链ID为键的map，存储了与不同区块链网络交互的服务实例，是实现多链支持的核心。

RankKey 等字段则用于特定的业务逻辑。

依赖集中管理：所有服务依赖都在这里创建和组装

多链支持：通过map存储不同链的服务实例

配置驱动：所有初始化都基于配置文件，便于部署

错误处理：使用 errors.Wrap 添加上下文信息

模块化：每个组件的初始化独立，便于维护和测试

可扩展：新增组件只需在上下文中添加字段并在本函数初始化

这个函数是整个应用的启动入口，完成后返回的 ServerCtx 将贯穿整个请求生命周期，为所有后续操作提供所需的基础设施。

go
// MintMetaNodeNFT 铸造MetaNodeNFT
// 接收铸造请求，与区块链交互，并返回交易结果
func MintMetaNodeNFT(ctx context.Context, svcCtx *svc.ServerCtx, req *typesv1.MetaNodeMintRequest) (*typesv1.MetaNodeMintResult, error) {
	// 记录开始铸造的日志
	xzap.WithContext(ctx).Info("Starting MetaNodeNFT mint",
		zap.Int("chain_id", req.ChainID),
		zap.String("to_address", req.ToAddress),
		zap.String("token_uri", req.TokenURI),
	)

	// ---------- 1. 获取并验证配置 ----------
	config, err := getMetaNodeConfig(svcCtx, req.ChainID)
	if err != nil {
		xzap.WithContext(ctx).Error("Failed to get MetaNode config", zap.Error(err))
		return nil, errors.Wrap(err, "failed to get MetaNode config")
	}
	// 记录加载的配置信息
	xzap.WithContext(ctx).Info("MetaNode config loaded",
		zap.String("contract_address", config.ContractAddress),
		zap.String("rpc_endpoint", config.RPCEndpoint),
		zap.Int64("chain_id", config.ChainID),
		zap.Uint64("gas_limit", config.GasLimit),
	)

	// ---------- 2. 验证请求参数 ----------
	if err := validateMetaNodeMintRequest(req); err != nil {
		xzap.WithContext(ctx).Error("Invalid mint request", zap.Error(err))
		return nil, errors.Wrap(err, "invalid mint request")
	}

	// ---------- 3. 连接区块链节点 ----------
	xzap.WithContext(ctx).Info("Connecting to Ethereum client", zap.String("rpc_endpoint", config.RPCEndpoint))
	client, err := ethclient.Dial(config.RPCEndpoint) // 建立与以太坊节点的RPC连接
	if err != nil {
		xzap.WithContext(ctx).Error("Failed to connect to Ethereum client", zap.Error(err))
		return nil, errors.Wrap(err, "failed to connect to Ethereum client")
	}
	defer client.Close() // 确保函数退出时关闭连接

	// 测试连接，获取节点返回的链ID
	chainID, err := client.ChainID(ctx)
	if err != nil {
		xzap.WithContext(ctx).Error("Failed to get chain ID from client", zap.Error(err))
		return nil, errors.Wrap(err, "failed to get chain ID")
	}
	// 确认连接的链与配置一致
	xzap.WithContext(ctx).Info("Connected to Ethereum client",
		zap.String("chain_id", chainID.String()),
		zap.Int64("expected_chain_id", config.ChainID),
	)

	// ---------- 4. 解析发送者账户 ----------
	// 从私钥字符串解析出ECDSA私钥对象
	privateKey, err := crypto.HexToECDSA(strings.TrimPrefix(config.OwnerPrivateKey, "0x"))
	if err != nil {
		return nil, errors.Wrap(err, "failed to parse private key")
	}
	// 从私钥推导出公钥
	publicKey := privateKey.Public()
	publicKeyECDSA, ok := publicKey.(*ecdsa.PublicKey)
	if !ok {
		return nil, errors.New("error casting public key to ECDSA")
	}
	// 从公钥推导出发送者地址
	fromAddress := crypto.PubkeyToAddress(*publicKeyECDSA)

	// ---------- 5. 获取交易参数 ----------
	// 获取账户的nonce（交易计数），用于防止重放攻击
	nonce, err := client.PendingNonceAt(ctx, fromAddress)
	if err != nil {
		return nil, errors.Wrap(err, "failed to get nonce")
	}

	// 确定Gas价格：优先使用配置的，否则从节点获取建议值
	gasPrice := config.GasPrice
	if gasPrice == nil {
		gasPrice, err = client.SuggestGasPrice(ctx)
		if err != nil {
			return nil, errors.Wrap(err, "failed to suggest gas price")
		}
	}

	// ---------- 6. 创建交易签名者 ----------
	// 创建交易授权对象，包含了签名者、链ID、Gas限制等所有交易参数
	auth, err := bind.NewKeyedTransactorWithChainID(privateKey, big.NewInt(config.ChainID))
	if err != nil {
		return nil, errors.Wrap(err, "failed to create transactor")
	}
	auth.Nonce = big.NewInt(int64(nonce))
	auth.Value = big.NewInt(0)          // 不发送ETH，纯铸造操作
	auth.GasLimit = config.GasLimit
	auth.GasPrice = gasPrice

	// ---------- 7. 准备调用合约 ----------
	// 解析合约ABI
	contractABI, err := abi.JSON(strings.NewReader(MetaNodeNFTABI))
	if err != nil {
		return nil, errors.Wrap(err, "failed to parse contract ABI")
	}
	// 创建绑定的合约实例
	contractAddress := common.HexToAddress(config.ContractAddress)
	contract := bind.NewBoundContract(contractAddress, contractABI, client, client, client)

	// ---------- 8. 调用合约方法 ----------
	toAddress := common.HexToAddress(req.ToAddress)
	// 记录即将发送的交易信息
	xzap.WithContext(ctx).Info("Calling safeMint function",
		zap.String("contract_address", config.ContractAddress),
		zap.String("to_address", req.ToAddress),
		zap.String("token_uri", req.TokenURI),
		zap.String("from_address", fromAddress.Hex()),
		zap.Uint64("nonce", nonce),
		zap.String("gas_price", gasPrice.String()),
		zap.Uint64("gas_limit", config.GasLimit),
	)

	// 执行交易，调用合约的safeMint方法
	tx, err := contract.Transact(auth, "safeMint", toAddress, req.TokenURI)
	if err != nil {
		xzap.WithContext(ctx).Error("Failed to call safeMint", zap.Error(err))
		return nil, errors.Wrap(err, "failed to call safeMint")
	}

	// 记录交易已发送
	xzap.WithContext(ctx).Info("MetaNodeNFT mint transaction sent",
		zap.String("tx_hash", tx.Hash().Hex()),
		zap.String("to_address", req.ToAddress),
		zap.String("token_uri", req.TokenURI),
		zap.String("contract_address", config.ContractAddress),
	)

	// ---------- 9. 等待交易确认 ----------
	// 等待交易被矿工打包进区块，最多等待30秒
	receipt, err := waitForTransaction(ctx, client, tx.Hash(), 30*time.Second)
	if err != nil {
		// 如果等待超时，返回pending状态，让客户端后续查询
		xzap.WithContext(ctx).Warn("transaction confirmation timeout", zap.Error(err))
		return &typesv1.MetaNodeMintResult{
			TxHash:   tx.Hash().Hex(),
			TokenID:  "",
			Status:   "pending",
			GasPrice: gasPrice.String(),
		}, nil
	}

	// ---------- 10. 从收据中解析TokenID ----------
	// 通过解析交易日志中的Transfer事件来获取铸造出的TokenID
	tokenID, err := parseTokenIDFromReceipt(receipt, contractABI)
	if err != nil {
		xzap.WithContext(ctx).Warn("failed to parse tokenID from receipt", zap.Error(err))
		tokenID = ""
	}

	// ---------- 11. 将铸造结果写入数据库 ----------
	// 链上操作成功后，将记录同步到本地数据库
	if tokenID != "" && getTransactionStatus(receipt) == "confirmed" {
		if err := insertMintedItemToDB(ctx, svcCtx, req.ChainID, config.ContractAddress, tokenID, req.ToAddress, req.Name); err != nil {
			// 数据库写入失败只记录错误，不影响铸造结果返回
			xzap.WithContext(ctx).Error("failed to insert minted item to database", zap.Error(err))
		} else {
			xzap.WithContext(ctx).Info("successfully inserted minted item to database",
				zap.String("chain_id", fmt.Sprintf("%d", req.ChainID)),
				zap.String("collection_address", config.ContractAddress),
				zap.String("token_id", tokenID),
				zap.String("owner", req.ToAddress),
			)
		}
	}

	// ---------- 12. 返回最终结果 ----------
	return &typesv1.MetaNodeMintResult{
		TxHash:      tx.Hash().Hex(),
		TokenID:     tokenID,
		Status:      getTransactionStatus(receipt),
		BlockNumber: receipt.BlockNumber.Int64(),
		GasUsed:     int64(receipt.GasUsed),
		GasPrice:    gasPrice.String(),
	}, nil
}

go
// processBids 处理并合并单品出价和集合出价，为每个TokenID生成最终出价记录
func processBids(tokenIds []string, itemsBestBids map[string]multi.Order, collectionBids []multi.Order, collectionAddr string) []types.ItemBid {
	// ---------- 1. 准备单品出价列表 ----------
	// 将单品最优出价map转换为切片，便于排序和遍历
	var itemsSortedBids []multi.Order
	for _, bid := range itemsBestBids {
		itemsSortedBids = append(itemsSortedBids, bid)
	}

	// 将单品出价按价格升序排序
	// 这样在后续处理中，价格低的单品出价会被优先处理
	sort.SliceStable(itemsSortedBids, func(i, j int) bool {
		return itemsSortedBids[i].Price.LessThan(itemsSortedBids[j].Price)
	})

	var resultBids []types.ItemBid
	var cBidIndex int // 集合出价的当前索引，用于遍历

	// ---------- 2. 第一轮处理：为没有单品出价的Token填充集合出价 ----------
	// 遍历原始的tokenIds列表（保持顺序）
	for _, tokenId := range tokenIds {
		// 如果该Token没有单品出价
		if _, ok := itemsBestBids[tokenId]; !ok {
			// 如果集合出价已经用完，跳过
			if cBidIndex >= len(collectionBids) {
				continue
			} else {
				// 否则，使用当前索引的集合出价作为该Token的出价
				resultBids = append(resultBids, types.ItemBid{
					MarketplaceId:     collectionBids[cBidIndex].MarketplaceId,
					CollectionAddress: collectionAddr,
					TokenId:           tokenId,
					OrderID:           collectionBids[cBidIndex].OrderID,
					EventTime:         collectionBids[cBidIndex].EventTime,
					ExpireTime:        collectionBids[cBidIndex].ExpireTime,
					Price:             collectionBids[cBidIndex].Price,
					Salt:              collectionBids[cBidIndex].Salt,
					BidSize:           collectionBids[cBidIndex].Size,
					BidUnfilled:       collectionBids[cBidIndex].QuantityRemaining,
					Bidder:            collectionBids[cBidIndex].Maker,
					OrderType:         getBidType(collectionBids[cBidIndex].OrderType),
				})
				cBidIndex++ // 移动到下一个集合出价
			}
		}
	}

	// ---------- 3. 第二轮处理：处理所有有单品出价的Token ----------
	// 遍历已排序的单品出价列表
	for _, itemBid := range itemsSortedBids {
		if cBidIndex >= len(collectionBids) {
			// 情况A：集合出价已用完，直接使用单品出价
			resultBids = append(resultBids, types.ItemBid{
				MarketplaceId:     itemBid.MarketplaceId,
				CollectionAddress: collectionAddr,
				TokenId:           itemBid.TokenId,
				OrderID:           itemBid.OrderID,
				EventTime:         itemBid.EventTime,
				ExpireTime:        itemBid.ExpireTime,
				Price:             itemBid.Price,
				Salt:              itemBid.Salt,
				BidSize:           itemBid.Size,
				BidUnfilled:       itemBid.QuantityRemaining,
				Bidder:            itemBid.Maker,
				OrderType:         getBidType(itemBid.OrderType),
			})
		} else {
			// 情况B：还有剩余的集合出价，需要比较价格
			cBid := collectionBids[cBidIndex]
			if cBid.Price.GreaterThan(itemBid.Price) {
				// 如果当前集合出价高于单品出价，使用集合出价
				resultBids = append(resultBids, types.ItemBid{
					MarketplaceId:     cBid.MarketplaceId,
					CollectionAddress: collectionAddr,
					TokenId:           itemBid.TokenId,
					OrderID:           cBid.OrderID,
					EventTime:         cBid.EventTime,
					ExpireTime:        cBid.ExpireTime,
					Price:             cBid.Price,
					Salt:              cBid.Salt,
					BidSize:           cBid.Size,
					BidUnfilled:       cBid.QuantityRemaining,
					Bidder:            cBid.Maker,
					OrderType:         getBidType(cBid.OrderType),
				})
				cBidIndex++ // 使用了集合出价，索引后移
			} else {
				// 如果单品出价不低于当前集合出价，使用单品出价
				resultBids = append(resultBids, types.ItemBid{
					MarketplaceId:     itemBid.MarketplaceId,
					CollectionAddress: collectionAddr,
					TokenId:           itemBid.TokenId,
					OrderID:           itemBid.OrderID,
					EventTime:         itemBid.EventTime,
					ExpireTime:        itemBid.ExpireTime,
					Price:             itemBid.Price,
					Salt:              itemBid.Salt,
					BidSize:           itemBid.Size,
					BidUnfilled:       itemBid.QuantityRemaining,
					Bidder:            itemBid.Maker,
					OrderType:         getBidType(itemBid.OrderType),
				})
				// 注意：这里没有移动cBidIndex，因为集合出价未被使用
			}
		}
	}

	return resultBids
}

算法目标：为输入的每个tokenId找到一个出价，优先级策略是：

优先使用该Token自身的出价（如果有）

如果没有自身出价，则使用集合级别的出价来填充

对于有自身出价的Token，用剩余的集合出价去替换那些价格更低的自身出价（提升整体出价质量）

场景：当用户查看一个NFT集合的多个物品时（例如BAYC系列的第1-10号），每个物品可能有：

自身出价：买家对特定#1234号的出价

集合出价：买家对整个BAYC系列任意一件的出价

展示需求：为每个物品展示一个“最相关”的出价，既要考虑单品特定出价，也要考虑集合出价，并且要公平地分配有限的集合出价资源。

算法优点：

确保每个tokenId都能得到一个出价（用集合出价填充空缺）

优先使用价格更高的出价

维持了输入tokenIds的顺序

// OFB模式的AES加密
func AesEncryptOFB(data []byte, key []byte) ([]byte, error) {
    // 1. PKCS7填充
    data = PKCS7Padding(data, aes.BlockSize)
    
    // 2. 创建AES加密块
    block, _ := aes.NewCipher([]byte(key))
    
    // 3. 分配输出空间（IV + 加密数据）
    out := make([]byte, aes.BlockSize+len(data))
    iv := out[:aes.BlockSize]
    
    // 4. 生成随机IV
    if _, err := io.ReadFull(rand.Reader, iv); err != nil {
        return nil, err
    }

    // 5. OFB模式加密
    stream := cipher.NewOFB(block, iv)
    stream.XORKeyStream(out[aes.BlockSize:], data)
    return out, nil
}

// PKCS7填充算法
func PKCS7Padding(ciphertext []byte, blocksize int) []byte {
    // 计算需要填充的字节数
    padding := blocksize - len(ciphertext)%blocksize
    // 生成填充字节（每个字节的值等于填充长度）
    padtext := bytes.Repeat([]byte{byte(padding)}, padding)
    return append(ciphertext, padtext...)
}

这是一个精心设计的批量查询函数，用于处理大量数据的场景：

go
func (d *Dao) QueryAllCollectionInfo(ctx context.Context, chain string) ([]multi.Collection, error) {
	// 1. 设置超时上下文，30秒后自动取消
	ctx, cancel := context.WithTimeout(ctx, QueryTimeout)
	defer cancel()

	// 2. 开启事务，确保整个读取过程的数据一致性
	tx := d.DB.WithContext(ctx).Begin()
	// 3. 使用defer处理panic情况，确保事务回滚
	defer func() {
		if r := recover(); r != nil {
			tx.Rollback()
			panic(r)
		}
	}()

	// 4. 初始化游标和结果集
	cursor := int64(0)
	var allCollections []multi.Collection
	
	// 5. 使用游标循环分页
	for {
		var collections []multi.Collection
		// 6. 带重试机制的查询
		for i := 0; i < MaxRetries; i++ {
			err := tx.Table(multi.CollectionTableName(chain)).
				Select(collectionFields).          // 只查询需要的字段
				Where("id > ?", cursor).           // 游标条件：只查询比当前最大ID大的记录
				Limit(MaxBatchReadCollections).    // 每批500条
				Order("id asc").                    // 按ID升序保证顺序
				Scan(&collections).Error

			if err == nil {
				break // 查询成功，跳出重试循环
			}
			// 如果最后一次重试仍然失败，回滚事务并返回错误
			if i == MaxRetries-1 {
				tx.Rollback()
				return nil, errors.Wrap(err, "failed on get collections info")
			}
			// 指数退避：等待时间随重试次数增加
			time.Sleep(time.Duration(i+1) * time.Second)
		}

		// 7. 收集当前批次的结果
		allCollections = append(allCollections, collections...)
		// 8. 如果返回数量少于批次大小，说明已到末尾
		if len(collections) < MaxBatchReadCollections {
			break
		}
		// 9. 更新游标为当前批次的最后一个ID
		cursor = collections[len(collections)-1].Id
	}

	// 10. 提交事务
	if err := tx.Commit().Error; err != nil {
		return nil, errors.Wrap(err, "failed to commit transaction")
	}
	return allCollections, nil
}
设计亮点：

游标分页：避免了传统 OFFSET 分页在大数据量下的性能问题

重试机制：对数据库波动有更好的容错性

超时控制：防止查询时间过长

事务包装：保证整个查询过程的数据一致性

func (d *Dao) QueryMultiChainUserCollectionInfos(ctx context.Context, chainID []int, chainNames []string, userAddrs []string) ([]types.UserCollections, error) {

		userAddrsParam += fmt.Sprintf(`'%s'`, addr)
		if i < len(userAddrs)-1 {
			userAddrsParam += ","
		}
	}

	// 2. SQL模板：外层SELECT * FROM (子查询) ORDER BY ...
	sqlHead := "SELECT * FROM ("
	sqlTail := ") as combined ORDER BY combined.floor_price * CAST(combined.item_count AS DECIMAL) DESC"
	var sqlMids []string

	// 3. 为每个链构建一个子查询
	for _, chainName := range chainNames {
		sqlMid := "("
		sqlMid += "select "
		sqlMid += "gc.address as address, gc.name as name, gc.floor_price as floor_price, " +
			"gc.chain_id as chain_id, gc.item_amount as item_amount, gc.symbol as symbol, " +
			"gc.image_uri as image_uri, count(*) as item_count "
		sqlMid += fmt.Sprintf("from %s as gc ", multi.CollectionTableName(chainName))
		sqlMid += fmt.Sprintf("join %s as gi ", multi.ItemTableName(chainName))
		sqlMid += "on gc.address = gi.collection_address "
		sqlMid += fmt.Sprintf("where gi.owner in (%s) ", userAddrsParam)
		sqlMid += "group by gc.address"
		sqlMid += ")"
		sqlMids = append(sqlMids, sqlMid)
	}

	// 4. 用UNION ALL组合所有子查询
	sql := sqlHead
	for i := 0; i < len(sqlMids); i++ {
		if i != 0 {
			sql += " UNION ALL "
		}
		sql += sqlMids[i]
	}
	sql += sqlTail

	// 5. 执行查询
	if err := d.DB.WithContext(ctx).Raw(sql).Scan(&userCollections).Error; err != nil {
		return nil, errors.Wrap(err, "failed on get user multi chain collection infos")
	}
	return userCollections, nil
}

func (d *Dao) QueryCollectionFloorChange(chain string, timeDiff int64) (map[string]float64, error) {
	collectionFloorChange := make(map[string]float64)
	var collectionPrices []multi.CollectionFloorPrice
	
	// 复杂的自查询：同时获取每个集合的最新地板价和时间点之前的地板价
	rawSql := fmt.Sprintf(`SELECT collection_address, price, event_time 
		FROM %s 
		WHERE (collection_address, event_time) IN (
			SELECT collection_address, MAX(event_time) 
			FROM %s 
			GROUP BY collection_address
		) OR (collection_address, event_time) IN (
			SELECT collection_address, MAX(event_time) 
			FROM %s 
			WHERE event_time <= UNIX_TIMESTAMP() - ? 
			GROUP BY collection_address
		) 
		ORDER BY collection_address,event_time DESC`, 
		multi.CollectionFloorPriceTableName(chain), 
		multi.CollectionFloorPriceTableName(chain), 
		multi.CollectionFloorPriceTableName(chain))
	
	if err := d.DB.Raw(rawSql, timeDiff).Scan(&collectionPrices).Error; err != nil {
		return nil, errors.Wrap(err, "failed on get collection floor change")
	}

	// 计算变化百分比：(当前 - 之前) / 之前
	for i := 0; i < len(collectionPrices); i++ {
		if i < len(collectionPrices)-1 && 
		   collectionPrices[i].CollectionAddress == collectionPrices[i+1].CollectionAddress && 
		   collectionPrices[i+1].Price.GreaterThan(decimal.Zero) {
			collectionFloorChange[collectionPrices[i].CollectionAddress] = 
				collectionPrices[i].Price.Sub(collectionPrices[i+1].Price).
				Div(collectionPrices[i+1].Price).InexactFloat64()
			i++
		} else {
			collectionFloorChange[collectionPrices[i].CollectionAddress] = 0.0
		}
	}
	return collectionFloorChange, nil
}

go
func (d *Dao) QueryCollectionItemOrder(ctx context.Context, chain string, filter types.CollectionItemFilterParams, collectionAddr string) ([]*CollectionItem, int64, error) {
	// 1. 默认市场处理
	if len(filter.Markets) == 0 {
		filter.Markets = []int{int(multi.OrderBookDex)}
	}

	db := d.DB.WithContext(ctx).Table(fmt.Sprintf("%s as ci", multi.ItemTableName(chain)))
	coTableName := multi.OrderTableName(chain)

	// 2. 根据状态条件分支查询（3种主要情况）
	// 情况1：单个状态（立即购买 或 有出价）
	if len(filter.Status) == 1 {
		db.Select("ci.id as id, ci.chain_id as chain_id, 0 as rarity_rank," +
			"ci.collection_address as collection_address,ci.token_id as token_id, ci.name as name, ci.owner as owner, " +
			"ci.is_opensea_banned as is_opensea_banned, " +
			"min(co.price) as list_price, SUBSTRING_INDEX(GROUP_CONCAT(co.marketplace_id ORDER BY co.price,co.marketplace_id),',', 1) AS market_id, min(co.price) != 0 as listing")

		// 立即购买分支
		if filter.Status[0] == BuyNow {
			db.Joins(fmt.Sprintf("join %s co on co.collection_address=ci.collection_address and co.token_id=ci.token_id", coTableName)).
				Where("co.collection_address = ? and co.order_type = ? and co.order_status=? and co.maker = ci.owner"+
					" and (ci.is_opensea_banned,co.marketplace_id)!=(true,1)",
					collectionAddr, multi.ListingOrder, multi.OrderStatusActive)
			// ... 市场过滤、TokenID过滤、用户地址过滤
			db.Group("co.token_id")
		}
		
		// 有出价分支
		if filter.Status[0] == HasOffer {
			// ... 类似逻辑，但查询出价订单
		}
	} 
	// 情况2：同时有挂单和出价
	else if len(filter.Status) == 2 {
		// 使用Having条件确保同时存在两种订单类型
		db.Group("co.token_id").Having("min(co.type)=? and max(co.type)=?", multi.ListingOrder, multi.OfferOrder)
	} 
	// 情况3：全部物品（无状态过滤）
	else {
		// 使用子查询关联挂单信息
		subQuery := d.DB.WithContext(ctx).Table(fmt.Sprintf("%s as cis", multi.ItemTableName(chain))).
			Select("cis.id as item_id,cis.collection_address as collection_address,cis.token_id as token_id, cis.owner as owner, cos.order_id as order_id, min(cos.price) as list_price, SUBSTRING_INDEX(GROUP_CONCAT(cos.marketplace_id ORDER BY cos.price,cos.marketplace_id),',', 1) AS market_id, min(cos.price) != 0 as listing").
			Joins(fmt.Sprintf("join %s cos on cos.collection_address=cis.collection_address and cos.token_id=cis.token_id", coTableName)).
			Where("cos.collection_address = ? and cos.order_type = ? and cos.order_status=? and cos.maker = cis.owner"+
				" and (cis.is_opensea_banned,cos.marketplace_id)!=(true,1)",
				collectionAddr, multi.ListingOrder, multi.OrderStatusActive)
		
		db.Joins("left join (?) co on co.collection_address=ci.collection_address and co.token_id=ci.token_id", subQuery)
	}

	// 3. 获取总数
	var count int64
	countTx := db.Session(&gorm.Session{})
	if err := countTx.Count(&count).Error; err != nil {
		return nil, 0, errors.Wrap(db.Error, "failed on count items")
	}

	// 4. 排序处理
	if filter.Sort == 0 {
		filter.Sort = listPriceAsc // 默认按价格升序
	}

	switch filter.Sort {
	case listTime:
		db.Order("list_time desc,ci.id asc")
	case listPriceAsc:
		db.Order("list_price asc, ci.id asc")
	case listPriceDesc:
		db.Order("list_price desc,ci.id asc")
	case salePriceDesc:
		db.Order("sale_price desc,ci.id asc")
	case salePriceAsc:
		db.Order("sale_price = 0,sale_price asc,ci.id asc")
	}

	// 5. 分页查询
	var items []*CollectionItem
	db.Offset(int((filter.Page - 1) * filter.PageSize)).Limit(int(filter.PageSize)).Scan(&items)

	return items, count, nil
}

go
func AuthMiddleWare(ctx *xkv.Store) gin.HandlerFunc {
	return func(c *gin.Context) {
		// 1. 从请求头获取 session_id
		values := c.Request.Header.Get("session_id")
		if values == "" {
			c.Next() // 没有session_id，直接放行（可能是公开接口）
			return
		}

		// 2. session_id 可能以逗号分隔（支持多会话）
		sessionIDs := strings.Split(values, ",")
		for _, sessionID := range sessionIDs {
			// 3. 将十六进制字符串解码为字节数组
			encryptCode, err := hex.DecodeString(sessionID)
			if err != nil {
				xhttp.Error(c, errcode.ErrTokenVerify) // 解码失败，返回token验证错误
				c.Abort()                               // 终止请求处理
				return
			}

			// 4. AES解密
			decrptCode, err := AesDecryptOFB(encryptCode, []byte(CR_LOGIN_SALT))
			if err != nil {
				xhttp.Error(c, errcode.ErrTokenExpire) // 解密失败，返回token过期
				c.Abort()
				return
			}

			// 5. 从Redis中验证会话
			result, err := ctx.Get(string(decrptCode))
			if result == "" || err != nil {
				xhttp.Error(c, errcode.ErrTokenExpire) // Redis中不存在或出错，返回token过期
				c.Abort()
				return
			}
		}

		// 6. 所有session_id验证通过，继续处理请求
		c.Next()
	}
}

func GetAuthUserAddress(c *gin.Context, ctx *xkv.Store) ([]string, error) {
	// 1. 获取session_id
	values := c.Request.Header.Get("session_id")
	if values == "" {
		return nil, errors.New("failed on get token")
	}

	sessionIDs := strings.Split(values, ",")
	var addrs []string
	for _, sessionID := range sessionIDs {
		// 2. 解码十六进制
		encryptCode, err := hex.DecodeString(sessionID)
		if err != nil {
			return nil, errors.Wrap(err, "failed on decode cookie")
		}

		// 3. AES解密
		decrptCode, err := AesDecryptOFB(encryptCode, []byte(CR_LOGIN_SALT))
		if err != nil {
			return nil, errors.Wrap(err, "invalid cookie")
		}

		// 4. 从Redis获取数据（与中间件重复，可能存在优化空间）
		result, err := ctx.Get(string(decrptCode))
		if result == "" || err != nil {
			return nil, errors.Wrap(err, "failed on read cookie from cache")
		}

		// 5. 解析解密后的内容，格式应为 "CR_LOGIN_KEY:用户地址"
		arr := strings.Split(string(decrptCode), CR_LOGIN_KEY+":")
		if len(arr) != 2 {
			return nil, errors.New("user cache info format err")
		}

		if arr[1] == "" {
			return nil, errors.New("invalid user address")
		}
		addrs = append(addrs, arr[1]) // 收集所有地址
	}

	return addrs, nil
}
🔐 AES解密函数
go
func AesDecryptOFB(data []byte, key []byte) ([]byte, error) {
	// 1. 创建AES密码块
	block, _ := aes.NewCipher([]byte(key))
	
	// 2. 提取初始化向量（IV），数据的前BlockSize字节
	iv := data[:aes.BlockSize]
	data = data[aes.BlockSize:]
	
	// 3. 验证密文长度
	if len(data)%aes.BlockSize != 0 {
		return nil, fmt.Errorf("data is not a multiple of the block size")
	}

	// 4. OFB模式解密
	out := make([]byte, len(data))
	mode := cipher.NewOFB(block, iv)
	mode.XORKeyStream(out, data)

	// 5. 去除PKCS7填充
	out = PKCS7UnPadding(out)
	return out, nil
}

// PKCS7UnPadding 去除填充
func PKCS7UnPadding(origData []byte) []byte {
	length := len(origData)
	unpadding := int(origData[length-1]) // 最后一个字节表示填充长度
	return origData[:(length - unpadding)]
}

1. 初始化变量

将它替换为 Gin 的 c.Writer

之后所有的响应写入都会同时进入原始ResponseWriter（给客户端）和这个buffer（供我们缓存）

3. 生成缓存Key

4. 尝试读取缓存
go
cacheData, err := (*store).Get(cacheKey)
if err == nil && cacheData != "" {
    cache := unserialize(cacheData)
    if cache != nil {
        // 4.1 恢复响应头
        for k, vals := range cache.Header {
            for _, v := range vals {
                bodyLogWriter.ResponseWriter.Header().Set(k, v)
            }
        }
        // 4.2 写入状态码
        bodyLogWriter.ResponseWriter.WriteHeader(cache.Status)
        
        // 4.3 检查是否是成功的响应（code=200）
        if err := json.Unmarshal(cache.Data, &data); err == nil {
            if data.Code == http.StatusOK {
                bodyLogWriter.ResponseWriter.Write(cache.Data)
                c.Abort()  // 直接返回，不再执行后续handler
                return
            }
        }
    }
}
这段代码做了几件事：

从Redis获取缓存数据

反序列化为 responseCache 对象

恢复响应头：把缓存中的Header重新设置

写入状态码：恢复HTTP状态码

验证响应：只有成功的响应（Code=200）才直接返回缓存

5. 继续执行请求
go
c.Next()

6. 获取响应体
go
responseBody := bodyLogWriter.body.Bytes()
handler执行完毕后，从之前设置的buffer中获取完整的响应内容。

7. 缓存成功响应

go
func RLog() gin.HandlerFunc {
	return func(c *gin.Context) {
		// ---------- 1. 记录请求信息 ----------
		// 保存原始路径和查询参数（因为后续中间件可能修改）
		path := c.Request.URL.Path
		query := c.Request.URL.RawQuery

		// ---------- 2. 捕获请求体 ----------
		// 使用 TeeReader 同时读取和保留请求体
		var buf bytes.Buffer
		tee := io.TeeReader(c.Request.Body, &buf)
		requestBody, _ := ioutil.ReadAll(tee)   // 读取请求体用于日志
		c.Request.Body = ioutil.NopCloser(&buf) // 恢复请求体供后续使用

		// ---------- 3. 替换 ResponseWriter 以捕获响应体 ----------
		bodyLogWriter := &BodyLogWriter{
			body:           bytes.NewBufferString(""),
			ResponseWriter: c.Writer,
		}
		c.Writer = bodyLogWriter

		// ---------- 4. 记录开始时间 ----------
		start := time.Now()

		// ---------- 5. 执行后续 handler ----------
		c.Next()

		// ---------- 6. 获取捕获的响应体 ----------
		responseBody := bodyLogWriter.body.Bytes()

		// ---------- 7. 准备日志记录器 ----------
		logger := xzap.WithContext(c.Request.Context())

		// ---------- 8. 错误处理 vs 正常日志 ----------
		if len(c.Errors) > 0 {
			// 8.1 如果有错误，记录所有错误
			for _, e := range c.Errors.Errors() {
				logger.Error(e)
			}
		} else {
			// 8.2 正常请求，记录详细日志
			latency := float64(time.Now().Sub(start).Nanoseconds() / 1000000.0) // 计算耗时（毫秒）
			
			fields := []zapcore.Field{
				zap.Int("status", c.Writer.Status()),           // HTTP状态码
				zap.String("method", c.Request.Method),         // 请求方法
				zap.String("function", c.HandlerName()),        // 处理函数名
				zap.String("path", path),                       // 请求路径
				zap.String("query", query),                     // 查询参数
				zap.String("ip", c.ClientIP()),                 // 客户端IP
				zap.String("user-agent", c.Request.UserAgent()), // User-Agent
				zap.String("token", c.Request.Header.Get("session_id")), // 会话ID
				zap.String("content-type", c.Request.Header.Get("Content-Type")), // 内容类型
				zap.Float64("latency", latency),                 // 处理耗时
				zap.String("request", string(requestBody)),      // 请求体
				zap.String("response", string(responseBody)),    // 响应体
			}
			logger.Info("Go-End", fields...)  // 记录信息级别日志
		}
	}
}

**4).recover**
这段代码实现了一个恐慌恢复中间件及其辅助函数，是 Gin 框架中用于保障服务稳定性的核心组件。它能够捕获处理请求过程中发生的 panic，防止程序崩溃，并记录详细的错误信息（包括堆栈跟踪）以便调试。

🛡️ 核心中间件：RecoverMiddleware
go
// RecoverMiddleware 恐慌捕获恢复处理
func RecoverMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		// 使用 defer 确保在发生 panic 时能够恢复
		defer func() {
			// 捕获 panic
			if cause := recover(); cause != nil {
				// 1. 记录错误日志：包含请求信息和堆栈
				xzap.WithContext(c.Request.Context()).Errorf(
					"[Recovery] panic recovered, request:%s%v [## stack:]:\n%s", 
					dumpRequest(c.Request), // 格式化请求信息
					cause,                   // panic 的原因
					dumpStack(3))            // 获取堆栈信息（跳过 3 层框架调用）
				
				// 2. 返回统一的错误响应给客户端
				xhttp.Error(c, errcode.ErrUnexpected)
			}
		}()

		// 继续执行后续的中间件和处理函数
		c.Next()
	}
}
工作流程：

defer 确保函数退出前执行恢复逻辑

recover() 捕获可能发生的 panic

记录详细的错误日志（请求信息 + 堆栈）

返回标准化的错误响应给客户端

服务继续运行，不会崩溃

dumpStack - 格式化堆栈
go
// dumpStack returns a nicely formatted stack frame, skipping skip frames.
func dumpStack(skip int) []byte {
	buf := new(bytes.Buffer)
	var lines [][]byte
	var lastFile string
	
	// 循环获取调用栈信息
	for i := skip; ; i++ {
		pc, file, line, ok := runtime.Caller(i) // 获取调用者信息
		if !ok {
			break
		}
		// 写入文件名、行号和程序计数器
		fmt.Fprintf(buf, "%s:%d (0x%x)\n", file, line, pc)
		
		// 读取源代码文件（用于显示具体代码行）
		if file != lastFile {
			data, _ := ioutil.ReadFile(file)
			lines = bytes.Split(data, []byte{'\n'})
			lastFile = file
		}
		// 写入函数名和对应的源代码行
		fmt.Fprintf(buf, "\t%s: %s\n", function(pc), source(lines, line))
	}
	return buf.Bytes()
}

// source 获取指定行的源代码内容
func source(lines [][]byte, n int) []byte {
	n-- // 堆栈中的行号从1开始，切片索引从0开始
	if n < 0 || n >= len(lines) {
		return dunno // 返回 "???" 表示无法获取
	}
	return bytes.TrimSpace(lines[n])
}

// function 获取函数名（去除包路径）
func function(pc uintptr) []byte {
	fn := runtime.FuncForPC(pc)
	if fn == nil {
		return dunno
	}
	name := []byte(fn.Name())
	// 去除路径部分，只保留函数名
	if lastslash := bytes.LastIndex(name, slash); lastslash >= 0 {
		name = name[lastslash+1:]
	}
	// 去除结构体接收者前缀
	if period := bytes.Index(name, dot); period >= 0 {
		name = name[period+1:]
	}
	return bytes.Replace(name, centerDot, dot, -1)
}

// dumpRequest 格式化请求样式，用于日志记录
func dumpRequest(req *http.Request) string {
	// 1. 复制请求体（因为只能读取一次）
	var dup io.ReadCloser
	req.Body, dup = dupReadCloser(req.Body)

	var b bytes.Buffer
	
	// 2. 写入请求行：方法 + URI + HTTP版本
	reqURI := req.RequestURI
	if reqURI == "" {
		reqURI = req.URL.RequestURI()
	}
	fmt.Fprintf(&b, "%s %s HTTP/%d.%d\n", req.Method, reqURI)

	// 3. 处理分块传输
	chunked := len(req.TransferEncoding) > 0 && req.TransferEncoding[0] == "chunked"
	if req.Body != nil {
		var dest io.Writer = &b
		if chunked {
			dest = httputil.NewChunkedWriter(dest)
		}
		n, _ := io.Copy(dest, req.Body)
		if chunked {
			dest.(io.Closer).Close()
		}
		if n > 0 {
			io.WriteString(&b, "\n")
		}
	}

	// 4. 恢复原始请求体
	req.Body = dup
	return b.String()
}

// dupReadCloser 复制请求体，一份用于读取，一份用于恢复
func dupReadCloser(reader io.ReadCloser) (io.ReadCloser, io.ReadCloser) {
	var buf bytes.Buffer
	tee := io.TeeReader(reader, &buf)
	return ioutil.NopCloser(tee), ioutil.NopCloser(&buf)
}

// getStacks 获取过滤后的调用堆栈信息
func getStacks() string {
	cs := stack.Trace().TrimBelow(stack.Caller(2)).TrimRuntime()
	var b strings.Builder

	for _, c := range cs {
		s := fmt.Sprintf("%+n\n\t%+v", c, c)
		// 过滤掉框架内部的堆栈，只保留业务代码
		if !strings.Contains(s, "net/http") && 
		   !strings.Contains(s, "gin-gonic/gin") && 
		   !strings.Contains(s, "go-playground/validator") {
			b.WriteString(s)
			b.WriteString("\n")
		}
	}
	return strings.TrimSpace(b.String())
}

func NewRouter(svcCtx *svc.ServerCtx) *gin.Engine {
	// 1. 创建 Gin 引擎并配置
	gin.ForceConsoleColor()                    // 强制控制台彩色输出
	gin.SetMode(gin.ReleaseMode)                // 设置为发布模式（减少日志，提升性能）
	r := gin.New()                              // 创建不带默认中间件的引擎

	// 2. 注册全局中间件（按顺序执行）
	r.Use(middleware.RecoverMiddleware())       // 恐慌恢复中间件（防止程序崩溃）
	r.Use(middleware.RLog())                     // 请求响应日志中间件

	// 3. CORS 跨域配置
	r.Use(cors.New(cors.Config{
		AllowAllOrigins:  true,                  // 允许所有来源（生产环境应限制）
		AllowMethods:     []string{"GET", "POST", "PUT", "DELETE", "OPTIONS", "PATCH"}, // 允许的HTTP方法
		AllowHeaders:     []string{"Origin", "Content-Length", "Content-Type", "X-CSRF-Token", "Authorization", "AccessToken", "Token"}, // 允许的请求头
		ExposeHeaders:    []string{"Content-Length", "Content-Type", "Access-Control-Allow-Origin", "Access-Control-Allow-Headers", "X-GW-Error-Code", "X-GW-Error-Message"}, // 暴露给客户端的响应头
		AllowCredentials: true,                   // 允许携带凭证（如Cookie）
		MaxAge:           1 * time.Hour,          // 预检请求的缓存时间
	}))

	// 4. 加载 v1 版本的路由（业务核心）
	loadV1(r, svcCtx)

	// 5. 设置 Swagger 文档
	r.GET("/swagger/*any", ginSwagger.WrapHandler(swaggerfiles.Handler))

	return r
}

		)

			xhttp.Error(c, errcode.NewCustomErr(err.Error()))
			return
		}

func TopRankingHandler(svcCtx *svc.ServerCtx) gin.HandlerFunc {
	return func(c *gin.Context) {
		// 1. 解析查询参数：limit 和 range
		limit, err := strconv.ParseInt(c.Query("limit"), 10, 64)
		if err != nil {
			xhttp.Error(c, errcode.ErrInvalidParams)
			return
		}

		// 2. 处理时间范围参数（有默认值）
		period := c.Query("range")
		if period != "" {
			// 验证 period 必须是预定义的值
			validParams := map[string]bool{
				"15m": true, "1h": true, "6h": true,
				"1d": true, "7d": true, "30d": true,
			}
			if ok := validParams[period]; !ok {
				xzap.WithContext(c).Error("range parse error: ", zap.String("range", period))
				xhttp.Error(c, errcode.ErrInvalidParams)
				return
			}
		} else {
			period = "1d" // 默认时间范围为1天
		}

		// 3. 并发查询所有链的排行榜数据
		var allResult []*types.CollectionRankingInfo
		var wg sync.WaitGroup
		var mu sync.Mutex

		// 为每个支持的链启动一个 goroutine
		for _, chain := range svcCtx.C.ChainSupported {
			wg.Add(1)
			go func(chain string) {
				defer wg.Done()

				// 调用服务层获取单个链的排行榜
				result, err := service.GetTopRanking(c.Copy(), svcCtx, chain, period, limit)
				if err != nil {
					xhttp.Error(c, err)
					return
				}

				// 使用互斥锁保护共享切片
				mu.Lock()
				allResult = append(allResult, result...)
				mu.Unlock()
			}(chain.Name)
		}

		// 等待所有 goroutine 完成
		wg.Wait()

		// 4. 合并后按交易量排序
		sort.SliceStable(allResult, func(i, j int) bool {
			return allResult[i].Volume.GreaterThan(allResult[j].Volume)
		})

		// 5. 返回最终排行榜
		xhttp.OkJson(c, types.CollectionRankingResp{Result: allResult})
	}
}
