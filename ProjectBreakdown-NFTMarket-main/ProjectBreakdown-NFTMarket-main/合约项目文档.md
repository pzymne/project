interface IEasySwapVault {
    // 定义金库接口，负责安全保管交易中的资产（ETH 和 NFT）
    // 这是订单簿系统的"保险箱"，确保资产安全直到交易完成

        OrderKey orderKey
    ) external view returns (uint256 ETHAmount, uint256 tokenId);
    // 查询订单的资产余额
    // view: 只读函数，不修改状态
    // 返回: 订单锁定的 ETH 数量和 NFT 的 tokenId
    
    function depositETH(OrderKey orderKey, uint256 ETHAmount) external payable;
    // 存入 ETH（创建 Bid 出价订单时调用）
    // payable: 接收 ETH 转账
    // 当用户创建购买订单时，需要锁定相应的 ETH
    
    function withdrawETH(
        OrderKey orderKey,
        uint256 ETHAmount,
        address to
    ) external;
    // 提取 ETH（订单取消或部分成交时调用）
    // 将锁定的 ETH 返还给用户
    
    function depositNFT(
        OrderKey orderKey,
        address from,
        address collection,
        uint256 tokenId
    ) external;
    // 存入 NFT（创建 List 挂单订单时调用）
    // 当用户创建出售订单时，需要将 NFT 转移到金库
    
    function withdrawNFT(
        OrderKey orderKey,
        address to,
        address collection,
        uint256 tokenId
    ) external;
    // 提取 NFT（订单取消时调用）
    // 将锁定的 NFT 返还给用户
    
    function editNFT(OrderKey oldOrderKey, OrderKey newOrderKey) external;
    // 编辑订单时转移 NFT 所有权
    // 当用户修改订单时，将 NFT 从旧订单转移到新订单
    
    function editETH(
        OrderKey oldOrderKey,
        OrderKey newOrderKey,
        uint256 oldETHAmount,
        uint256 newETHAmount,
        address to
    ) external payable;
    // 编辑订单时调整 ETH
    // payable: 如果新价格更高，需要补充 ETH
    // 如果新价格更低，退还部分 ETH
    
    function batchTransferERC721(
        address to,
        LibOrder.NFTInfo[] calldata assets
    ) external;
    // 批量转移 NFT（用于批量成交）
    // 提高 gas 效率，减少多次单独转移的成本

        address from,
        address to,
        LibOrder.Asset calldata assets
    ) external;
    // 单个 NFT 转移（用于普通成交）

    /**
     * @notice 批量创建订单并转移相关资产
     * @param newOrders 多个订单结构数据数组
     * @return newOrderKeys 返回订单唯一标识数组
     */
    function makeOrders(
        LibOrder.Order[] calldata newOrders  // calldata比memory更省gas
    )
        external
        payable  // 可以接收ETH
        override  // 实现接口
        whenNotPaused  // 合约未暂停时才能调用
        nonReentrant  // 防止重入攻击
        returns (OrderKey[] memory newOrderKeys)
    {
        uint256 orderAmount = newOrders.length;  // 获取订单数量
        newOrderKeys = new OrderKey[](orderAmount);  // 初始化返回数组

        uint128 ETHAmount;  // 累计需要的ETH总金额
        for (uint256 i = 0; i < orderAmount; ++i) {  // 遍历所有订单
            uint128 buyPrice;  // Bid订单的出价金额
            if (newOrders[i].side == LibOrder.Side.Bid) {  // 如果是Bid订单
                // 计算Bid订单需要的ETH：单价 × 数量
                buyPrice =
                    Price.unwrap(newOrders[i].price) *  // 获取价格数值
                    newOrders[i].nft.amount;  // NFT数量
            }

            // 尝试创建订单
            OrderKey newOrderKey = _makeOrderTry(newOrders[i], buyPrice);
            newOrderKeys[i] = newOrderKey;  // 保存结果
            if (
                // 如果订单创建成功，累计ETH金额
                OrderKey.unwrap(newOrderKey) !=
                OrderKey.unwrap(LibOrder.ORDERKEY_SENTINEL)  // 检查是否为空标识
            ) {
                ETHAmount += buyPrice;  // 累计ETH金额
            }
        }

        if (msg.value > ETHAmount) {
            // 如果传入的ETH多于实际需要的金额，退回多余的ETH
            _msgSender().safeTransferETH(msg.value - ETHAmount);
        }
    }

    /**
     * @notice 批量取消订单
     * @param orderKeys 要取消的订单唯一标识数组
     * @return successes 返回每个订单的取消结果
     */
    function cancelOrders(
        OrderKey[] calldata orderKeys
    )
        external
        override
        whenNotPaused
        nonReentrant
        returns (bool[] memory successes)
    {
        successes = new bool[](orderKeys.length);  // 初始化结果数组

        for (uint256 i = 0; i < orderKeys.length; ++i) {
            bool success = _cancelOrderTry(orderKeys[i]);  // 尝试取消订单
            successes[i] = success;  // 保存结果
        }
    }

    /**
     * @notice 批量编辑订单（修改订单价格）
     * @param editDetails 编辑详情数组
     * @return newOrderKeys 返回新订单的唯一标识数组
     */
    function editOrders(
        LibOrder.EditDetail[] calldata editDetails
    )
        external
        payable
        override
        whenNotPaused
        nonReentrant
        returns (OrderKey[] memory newOrderKeys)
    {
        newOrderKeys = new OrderKey[](editDetails.length);  // 初始化返回数组

        uint256 bidETHAmount;  // Bid订单需要补足的ETH总金额
        for (uint256 i = 0; i < editDetails.length; ++i) {
            // 尝试编辑订单
            (OrderKey newOrderKey, uint256 bidPrice) = _editOrderTry(
                editDetails[i].oldOrderKey,
                editDetails[i].newOrder
            );
            bidETHAmount += bidPrice;  // 累计需要补足的ETH
            newOrderKeys[i] = newOrderKey;  // 保存结果
        }

        if (msg.value > bidETHAmount) {
            // 如果传入的ETH多于实际需要的金额，退回多余的ETH
            _msgSender().safeTransferETH(msg.value - bidETHAmount);
        }
    }

    /**
     * @notice 匹配单个订单（撮合交易）
     * @param sellOrder 挂单订单（List订单）
     * @param buyOrder 出价订单（Bid订单）
     */
    function matchOrder(
        LibOrder.Order calldata sellOrder,
        LibOrder.Order calldata buyOrder
    ) external payable override whenNotPaused nonReentrant {
        uint256 costValue = _matchOrder(sellOrder, buyOrder, msg.value);  // 执行匹配
        if (msg.value > costValue) {
            // 如果传入的ETH多于实际需要的金额，退回多余的ETH
            _msgSender().safeTransferETH(msg.value - costValue);
        }
    }

    /**
     * @notice 批量原子性匹配订单（批量撮合交易）
     * @param matchDetails 匹配详情数组
     * @return successes 返回每个订单的匹配结果
     */
    /// @custom:oz-upgrades-unsafe-allow delegatecall
    function matchOrders(
        LibOrder.MatchDetail[] calldata matchDetails
    )
        external
        payable
        override
        whenNotPaused
        nonReentrant
        returns (bool[] memory successes)
    {
        successes = new bool[](matchDetails.length);  // 初始化结果数组

        uint128 buyETHAmount;  // 累计买家已花费的ETH金额

        for (uint256 i = 0; i < matchDetails.length; ++i) {
            LibOrder.MatchDetail calldata matchDetail = matchDetails[i];
            // 使用delegatecall调用内部匹配函数，保持相同的存储上下文
            (bool success, bytes memory data) = address(this).delegatecall(
                abi.encodeWithSignature(  // 编码函数签名和参数
                    "matchOrderWithoutPayback((uint8,uint8,address,(uint256,address,uint96),uint128,uint64,uint64),(uint8,uint8,address,(uint256,address,uint96),uint128,uint64,uint64),uint256)",
                    matchDetail.sellOrder,
                    matchDetail.buyOrder,
                    msg.value - buyETHAmount  // 剩余可用ETH
                )
            );

            if (success) {
                successes[i] = success;  // 保存成功结果
                if (matchDetail.buyOrder.maker == _msgSender()) {
                    // 如果是买家发起的匹配，累计已花费的ETH
                    uint128 buyPrice;
                    buyPrice = abi.decode(data, (uint128));  // 解码返回数据
                    buyETHAmount += buyPrice;  // 累计花费
                }
            } else {
                // 记录批量匹配中的错误
                emit BatchMatchInnerError(i, data);
            }
        }

        if (msg.value > buyETHAmount) {
            // 如果传入的ETH多于实际需要的金额，退回多余的ETH
            _msgSender().safeTransferETH(msg.value - buyETHAmount);
        }
    }

    /**
     * @notice 匹配订单但不退回多余ETH（内部函数，仅用于批量匹配）
     * @param sellOrder 挂单订单
     * @param buyOrder 出价订单
     * @param msgValue 传入的ETH金额
     * @return costValue 实际花费的ETH金额
     */
    function matchOrderWithoutPayback(
        LibOrder.Order calldata sellOrder,
        LibOrder.Order calldata buyOrder,
        uint256 msgValue
    )
        external
        payable
        whenNotPaused
        onlyDelegateCall  // 只能通过delegatecall调用
        returns (uint128 costValue)
    {
        costValue = _matchOrder(sellOrder, buyOrder, msgValue);  // 执行匹配
    }

    /**
     * @notice 尝试创建订单（内部函数）
     * @param order 要创建的订单
     * @param ETHAmount Bid订单需要的ETH金额
     * @return newOrderKey 订单唯一标识
     */
    function _makeOrderTry(
        LibOrder.Order calldata order,
        uint128 ETHAmount
    ) internal returns (OrderKey newOrderKey) {
        // 验证订单有效性
        if (
            order.maker == _msgSender() &&  // 只有订单创建者可以创建订单
            Price.unwrap(order.price) != 0 &&  // 价格不能为0
            order.salt != 0 &&  // salt不能为0
            (order.expiry > block.timestamp || order.expiry == 0) &&  // 过期时间必须有效
            filledAmount[LibOrder.hash(order)] == 0  // 订单不能已被取消或完全成交
        ) {
            newOrderKey = LibOrder.hash(order);  // 计算订单哈希作为唯一标识

            // 根据订单类型处理资产
            if (order.side == LibOrder.Side.List) {  // List订单
                if (order.nft.amount != 1) {
                    // List订单限制数量为1
                    return LibOrder.ORDERKEY_SENTINEL;  // 返回空标识
                }
                // 将NFT存入金库
                IEasySwapVault(_vault).depositNFT(
                    newOrderKey,
                    order.maker,
                    order.nft.collection,
                    order.nft.tokenId
                );
            } else if (order.side == LibOrder.Side.Bid) {  // Bid订单
                if (order.nft.amount == 0) {
                    // Bid订单数量不能为0
                    return LibOrder.ORDERKEY_SENTINEL;  // 返回空标识
                }
                // 将ETH存入金库
                IEasySwapVault(_vault).depositETH{value: uint256(ETHAmount)}(
                    newOrderKey,
                    ETHAmount
                );
            }

            // 将订单添加到订单存储
            _addOrder(order);

            // 发出订单创建事件
            emit LogMake(
                newOrderKey,
                order.side,
                order.saleKind,
                order.maker,
                order.nft,
                order.price,
                order.expiry,
                order.salt
            );
        } else {
            // 订单创建失败，发出跳过事件
            emit LogSkipOrder(LibOrder.hash(order), order.salt);
        }
    }

    /**
     * @notice 尝试取消订单（内部函数）
     * @param orderKey 要取消的订单唯一标识
     * @return success 是否成功取消
     */
    function _cancelOrderTry(
        OrderKey orderKey
    ) internal returns (bool success) {
        LibOrder.Order memory order = orders[orderKey].order;  // 获取订单信息

        if (
            order.maker == _msgSender() &&  // 只有订单创建者可以取消
            filledAmount[orderKey] < order.nft.amount  // 只有未完全成交的订单才能取消
        ) {
            OrderKey orderHash = LibOrder.hash(order);  // 计算订单哈希
            
            // 从订单存储中移除订单
            _removeOrder(order);
            
            // 根据订单类型提取资产
            if (order.side == LibOrder.Side.List) {  // List订单
                // 提取NFT返回给创建者
                IEasySwapVault(_vault).withdrawNFT(
                    orderHash,
                    order.maker,
                    order.nft.collection,
                    order.nft.tokenId
                );
            } else if (order.side == LibOrder.Side.Bid) {  // Bid订单
                // 计算未成交数量
                uint256 availNFTAmount = order.nft.amount -
                    filledAmount[orderKey];
                // 提取对应ETH返回给创建者
                IEasySwapVault(_vault).withdrawETH(
                    orderHash,
                    Price.unwrap(order.price) * availNFTAmount,  // 提取金额 = 单价 × 未成交数量
                    order.maker
                );
            }
            // 标记订单为已取消
            _cancelOrder(orderKey);
            success = true;  // 设置成功标志
            emit LogCancel(orderKey, order.maker);  // 发出取消事件
        } else {
            // 取消失败，发出跳过事件
            emit LogSkipOrder(orderKey, order.salt);
        }
    }

    /**
     * @notice 尝试编辑订单（内部函数）
     * @param oldOrderKey 旧订单唯一标识
     * @param newOrder 新订单信息
     * @return newOrderKey 新订单唯一标识
     * @return deltaBidPrice Bid订单需要补足的ETH金额
     */
    function _editOrderTry(
        OrderKey oldOrderKey,
        LibOrder.Order calldata newOrder
    ) internal returns (OrderKey newOrderKey, uint256 deltaBidPrice) {
        LibOrder.Order memory oldOrder = orders[oldOrderKey].order;  // 获取旧订单

        // 检查订单编辑限制：只能修改价格和数量
        if (
            (oldOrder.saleKind != newOrder.saleKind) ||  // saleKind不能改变
            (oldOrder.side != newOrder.side) ||  // side不能改变
            (oldOrder.maker != newOrder.maker) ||  // maker不能改变
            (oldOrder.nft.collection != newOrder.nft.collection) ||  // NFT集合不能改变
            (oldOrder.nft.tokenId != newOrder.nft.tokenId) ||  // tokenId不能改变
            filledAmount[oldOrderKey] >= oldOrder.nft.amount  // 订单不能已完全成交
        ) {
            emit LogSkipOrder(oldOrderKey, oldOrder.salt);
            return (LibOrder.ORDERKEY_SENTINEL, 0);  // 返回空标识
        }

        // 检查新订单是否有效
        if (
            newOrder.maker != _msgSender() ||  // 新订单的maker必须是调用者
            newOrder.salt == 0 ||  // salt不能为0
            (newOrder.expiry < block.timestamp && newOrder.expiry != 0) ||  // 过期时间必须有效
            filledAmount[LibOrder.hash(newOrder)] != 0  // 新订单不能已被取消或完全成交
        ) {
            emit LogSkipOrder(oldOrderKey, newOrder.salt);
            return (LibOrder.ORDERKEY_SENTINEL, 0);  // 返回空标识
        }

        // 取消旧订单
        uint256 oldFilledAmount = filledAmount[oldOrderKey];  // 获取旧订单已成交数量
        _removeOrder(oldOrder);  // 从订单存储中移除
        _cancelOrder(oldOrderKey);  // 从订单簿中取消
        emit LogCancel(oldOrderKey, oldOrder.maker);  // 发出取消事件

        // 创建新订单
        newOrderKey = _addOrder(newOrder);  // 添加到订单存储

        // 更新金库中的资产关联
        if (oldOrder.side == LibOrder.Side.List) {  // List订单
            // 直接更新NFT关联
            IEasySwapVault(_vault).editNFT(oldOrderKey, newOrderKey);
        } else if (oldOrder.side == LibOrder.Side.Bid) {  // Bid订单
            // 计算价格差额
            uint256 oldRemainingPrice = Price.unwrap(oldOrder.price) *
                (oldOrder.nft.amount - oldFilledAmount);  // 旧订单剩余价格
            uint256 newRemainingPrice = Price.unwrap(newOrder.price) *
                newOrder.nft.amount;  // 新订单总价格
                
            if (newRemainingPrice > oldRemainingPrice) {
                // 新价格更高，需要补足差额
                deltaBidPrice = newRemainingPrice - oldRemainingPrice;
                IEasySwapVault(_vault).editETH{value: uint256(deltaBidPrice)}(
                    oldOrderKey,
                    newOrderKey,
                    oldRemainingPrice,
                    newRemainingPrice,
                    oldOrder.maker
                );
            } else {
                // 新价格更低或相等，金库会退回多余ETH
                IEasySwapVault(_vault).editETH(
                    oldOrderKey,
                    newOrderKey,
                    oldRemainingPrice,
                    newRemainingPrice,
                    oldOrder.maker
                );
            }
        }

        // 发出新订单创建事件
        emit LogMake(
            newOrderKey,
            newOrder.side,
            newOrder.saleKind,
            newOrder.maker,
            newOrder.nft,
            newOrder.price,
            newOrder.expiry,
            newOrder.salt
        );
    }

    /**
     * @notice 匹配订单的核心逻辑（内部函数）
     * @param sellOrder 挂单订单
     * @param buyOrder 出价订单
     * @param msgValue 调用者传入的ETH金额
     * @return costValue 实际花费的ETH金额
     */
    function _matchOrder(
        LibOrder.Order calldata sellOrder,
        LibOrder.Order calldata buyOrder,
        uint256 msgValue
    ) internal returns (uint128 costValue) {
        OrderKey sellOrderKey = LibOrder.hash(sellOrder);  // 计算卖单哈希
        OrderKey buyOrderKey = LibOrder.hash(buyOrder);    // 计算买单哈希
        
        // 检查订单是否可匹配
        _isMatchAvailable(sellOrder, buyOrder, sellOrderKey, buyOrderKey);

        if (_msgSender() == sellOrder.maker) {  // 场景1：卖家接受出价
            require(msgValue == 0, "HD: value > 0");  // 卖家接受出价时不需要传入ETH
            
            bool isSellExist = orders[sellOrderKey].order.maker != address(0);  // 检查sellOrder是否存在于订单存储中
            _validateOrder(sellOrder, isSellExist);  // 验证sellOrder
            _validateOrder(orders[buyOrderKey].order, false);  // 验证buyOrder

            uint128 fillPrice = Price.unwrap(buyOrder.price);  // 成交价格为Bid订单的价格
            
            if (isSellExist) {
                // 如果sellOrder存在于订单存储中，移除并标记为完全成交
                _removeOrder(sellOrder);
                _updateFilledAmount(sellOrder.nft.amount, sellOrderKey);  // sellOrder完全成交
            }
            // 更新buyOrder的成交数量
            _updateFilledAmount(filledAmount[buyOrderKey] + 1, buyOrderKey);
            
            emit LogMatch(  // 发出匹配事件
                sellOrderKey,
                buyOrderKey,
                sellOrder,
                buyOrder,
                fillPrice
            );

            // 资产转移
            // 1. 从金库提取buyOrder的ETH到合约
            IEasySwapVault(_vault).withdrawETH(
                buyOrderKey,
                fillPrice,
                address(this)
            );

            // 2. 计算协议手续费并转给卖家（扣除手续费后的金额）
            uint128 protocolFee = _shareToAmount(fillPrice, protocolShare);  // 计算手续费
            sellOrder.maker.safeTransferETH(fillPrice - protocolFee);  // 转给卖家

            // 3. 转移NFT给买家
            if (isSellExist) {
                // 如果sellOrder存在于订单存储中，从金库提取NFT
                IEasySwapVault(_vault).withdrawNFT(
                    sellOrderKey,
                    buyOrder.maker,
                    sellOrder.nft.collection,
                    sellOrder.nft.tokenId
                );
            } else {
                // 如果sellOrder不存在于订单存储中，直接从卖家转移NFT给买家
                IEasySwapVault(_vault).transferERC721(
                    sellOrder.maker,
                    buyOrder.maker,
                    sellOrder.nft
                );
            }
        } else if (_msgSender() == buyOrder.maker) {  // 场景2：买家接受挂单
            bool isBuyExist = orders[buyOrderKey].order.maker != address(0);  // 检查buyOrder是否存在于订单存储中
            _validateOrder(orders[sellOrderKey].order, false);  // 验证sellOrder
            _validateOrder(buyOrder, isBuyExist);  // 验证buyOrder

            uint128 buyPrice = Price.unwrap(buyOrder.price);  // 买家的出价
            uint128 fillPrice = Price.unwrap(sellOrder.price);  // 成交价格为List订单的价格
            
            if (!isBuyExist) {  // buyOrder不存在于订单存储中
                require(msgValue >= fillPrice, "HD: value < fill price");  // 需要传入足够的ETH
            } else {  // buyOrder存在于订单存储中
                require(buyPrice >= fillPrice, "HD: buy price < fill price");  // 验证出价是否足够
                // 从金库提取buyOrder的ETH到合约
                IEasySwapVault(_vault).withdrawETH(
                    buyOrderKey,
                    buyPrice,
                    address(this)
                );
                // 移除buyOrder并标记为完全成交
                _removeOrder(buyOrder);
                _updateFilledAmount(filledAmount[buyOrderKey] + 1, buyOrderKey);
            }
            // 标记sellOrder为完全成交
            _updateFilledAmount(sellOrder.nft.amount, sellOrderKey);

            emit LogMatch(  // 发出匹配事件
                buyOrderKey,
                sellOrderKey,
                buyOrder,
                sellOrder,
                fillPrice
            );

            // 资产转移
            // 1. 计算协议手续费并转给卖家（扣除手续费后的金额）
            uint128 protocolFee = _shareToAmount(fillPrice, protocolShare);
            sellOrder.maker.safeTransferETH(fillPrice - protocolFee);
            
            // 2. 如果买家出价高于成交价，退回多余ETH
            if (buyPrice > fillPrice) {
                buyOrder.maker.safeTransferETH(buyPrice - fillPrice);
            }

            // 3. 从金库提取NFT给买家
            IEasySwapVault(_vault).withdrawNFT(
                sellOrderKey,
                buyOrder.maker,
                sellOrder.nft.collection,
                sellOrder.nft.tokenId
            );
            
            // 返回实际花费的ETH金额
            costValue = isBuyExist ? 0 : buyPrice;  // 如果buyOrder已存在于订单存储中，则不需要额外ETH
        } else {
            revert("HD: sender invalid");  // 调用者无效
        }
    }

    /**
     * @notice 检查订单是否可匹配（内部函数）
     * @param sellOrder 挂单订单
     * @param buyOrder 出价订单
     * @param sellOrderKey 挂单订单唯一标识
     * @param buyOrderKey 出价订单唯一标识
     */
    function _isMatchAvailable(
        LibOrder.Order memory sellOrder,
        LibOrder.Order memory buyOrder,
        OrderKey sellOrderKey,
        OrderKey buyOrderKey
    ) internal view {
        require(
            OrderKey.unwrap(sellOrderKey) != OrderKey.unwrap(buyOrderKey),
            "HD: same order"  // 两个订单不能是同一个订单
        );
        require(
            sellOrder.side == LibOrder.Side.List &&  // sellOrder必须是List订单
                buyOrder.side == LibOrder.Side.Bid,  // buyOrder必须是Bid订单
            "HD: side mismatch"
        );
        require(
            sellOrder.saleKind == LibOrder.SaleKind.FixedPriceForItem,  // sellOrder必须是固定价格类型
            "HD: kind mismatch"
        );
        require(sellOrder.maker != buyOrder.maker, "HD: same maker");  // 两个订单的创建者不能是同一人
        require( // 检查资产是否匹配
            buyOrder.saleKind == LibOrder.SaleKind.FixedPriceForCollection ||  // buyOrder可以是集合出价
                (sellOrder.nft.collection == buyOrder.nft.collection &&  // 或者资产完全匹配
                    sellOrder.nft.tokenId == buyOrder.nft.tokenId),
            "HD: asset mismatch"
        );
        require(
            filledAmount[sellOrderKey] < sellOrder.nft.amount &&  // sellOrder未完全成交
                filledAmount[buyOrderKey] < buyOrder.nft.amount,  // buyOrder未完全成交
            "HD: order closed"
        );
    }

    /**
     * @notice 根据手续费比例计算手续费金额
     * @param total 总金额
     * @param share 手续费比例（基点制）
     * @return 计算出的手续费金额
     */
    function _shareToAmount(
        uint128 total,
        uint128 share
    ) internal pure returns (uint128) {
        return (total * share) / LibPayInfo.TOTAL_SHARE;  // 按比例计算手续费
    }

    /**
     * @notice 检查是否为delegatecall调用
     */
    function _checkDelegateCall() private view {
        require(address(this) != self);  // 当前地址不等于原始地址
    }

    /**
     * @notice 设置金库合约地址（仅所有者）
     * @param newVault 新的金库合约地址
     */
    function setVault(address newVault) public onlyOwner {
        require(newVault != address(0), "HD: zero address");  // 地址不能为零
        _vault = newVault;
    }

    /**
     * @notice 提取ETH（仅所有者，用于提取协议手续费）
     * @param recipient 接收地址
     * @param amount 提取金额
     */
    function withdrawETH(
        address recipient,
        uint256 amount
    ) external nonReentrant onlyOwner {
        recipient.safeTransferETH(amount);  // 安全转账ETH
        emit LogWithdrawETH(recipient, amount);  // 发出提取事件
    }

    /**
     * @notice 暂停合约（仅所有者）
     */
    function pause() external onlyOwner {
        _pause();  // 暂停合约
    }

    /**
     * @notice 恢复合约（仅所有者）
     */
    function unpause() external onlyOwner {
        _unpause();  // 恢复合约
    }

    /**
     * @notice 接收ETH的回退函数
     * @dev 用于接收Bid订单的ETH和协议手续费
     */
    receive() external payable {}  // 允许合约接收ETH

    /**
     * @notice 可升级合约的存储间隙
     * @dev 为未来升级预留存储空间
     */
    uint256[50] private __gap;  // 存储间隙，防止存储冲突
}

pragma solidity ^0.8.19;
// 指定Solidity编译器版本为0.8.19及以上

// 导入OpenZeppelin可升级合约的权限管理模块
import {OwnableUpgradeable} from "@openzeppelin/contracts-upgradeable/access/OwnableUpgradeable.sol";

// 导入自定义安全转账库和ERC721接口
import {LibTransferSafeUpgradeable, IERC721} from "./libraries/LibTransferSafeUpgradeable.sol";
// 导入订单相关定义和类型
import {LibOrder, OrderKey} from "./libraries/LibOrder.sol";

// 导入金库接口定义
import {IEasySwapVault} from "./interface/IEasySwapVault.sol";

// 定义EasySwapVault合约，实现金库接口，并继承可升级的所有权管理
contract EasySwapVault is IEasySwapVault, OwnableUpgradeable {
    // 使用库扩展address类型的函数，添加安全转账功能
    using LibTransferSafeUpgradeable for address;
    // 使用库扩展IERC721接口的函数，添加安全的NFT转移功能
    using LibTransferSafeUpgradeable for IERC721;

    // 公共状态变量：订单簿合约地址
    address public orderBook;
    // 映射：订单键 → 锁定的ETH余额
    mapping(OrderKey => uint256) public ETHBalance;
    // 映射：订单键 → 锁定的NFT tokenId
    mapping(OrderKey => uint256) public NFTBalance;

    // 修饰符：仅允许EasySwap订单簿合约调用
    modifier onlyEasySwapOrderBook() {
        // 验证调用者是否为订单簿合约
        require(msg.sender == orderBook, "HV: only EasySwap OrderBook");
        _; // 执行原函数体
    }

    // 初始化函数（可升级合约的初始化入口）
    function initialize() public initializer {
        // 调用父合约OwnableUpgradeable的初始化函数，将部署者设为所有者
        __Ownable_init(_msgSender());
    }

    // 设置订单簿合约地址（仅所有者可调用）
    function setOrderBook(address newOrderBook) public onlyOwner {
        // 验证新地址不能为零地址
        require(newOrderBook != address(0), "HV: zero address");
        // 更新订单簿地址
        orderBook = newOrderBook;
    }

    // 查询订单的资产余额
    function balanceOf(
        OrderKey orderKey // 订单唯一标识
    ) external view returns (uint256 ETHAmount, uint256 tokenId) {
        // 返回订单锁定的ETH金额
        ETHAmount = ETHBalance[orderKey];
        // 返回订单锁定的NFT tokenId（如果为0表示没有NFT）
        tokenId = NFTBalance[orderKey];
    }

    // 存入ETH（创建Bid订单时调用）
    function depositETH(
        OrderKey orderKey, // 订单标识
        uint256 ETHAmount  // 预期存入的ETH金额
    ) external payable onlyEasySwapOrderBook {
        // 验证实际发送的ETH不小于预期金额
        require(msg.value >= ETHAmount, "HV: not match ETHAmount");
        // 累加订单的ETH余额（使用msg.value而非ETHAmount，可能接受超额支付）
        ETHBalance[orderKey] += msg.value;
    }

    // 提取ETH（订单取消或匹配时调用）
    function withdrawETH(
        OrderKey orderKey, // 订单标识
        uint256 ETHAmount, // 提取的ETH金额
        address to         // 接收ETH的地址
    ) external onlyEasySwapOrderBook {
        // 从订单余额中扣除提取金额
        ETHBalance[orderKey] -= ETHAmount;
        // 安全地将ETH转账给接收者
        to.safeTransferETH(ETHAmount);
    }

    // 存入NFT（创建List订单时调用）
    function depositNFT(
        OrderKey orderKey,   // 订单标识
        address from,        // NFT当前所有者（卖家）
        address collection,  // NFT合约地址
        uint256 tokenId      // NFT的tokenId
    ) external onlyEasySwapOrderBook {
        // 安全地将NFT从所有者转移到金库合约
        IERC721(collection).safeTransferNFT(from, address(this), tokenId);

        // 记录NFT归属到指定订单
        NFTBalance[orderKey] = tokenId;
    }

    // 提取NFT（订单取消或匹配时调用）
    function withdrawNFT(
        OrderKey orderKey,   // 订单标识
        address to,          // 接收NFT的地址
        address collection,  // NFT合约地址
        uint256 tokenId      // NFT的tokenId
    ) external onlyEasySwapOrderBook {
        // 验证金库中该订单确实持有指定的NFT
        require(NFTBalance[orderKey] == tokenId, "HV: not match tokenId");
        // 清除订单的NFT记录
        delete NFTBalance[orderKey];

        // 安全地将NFT从金库转移到接收者
        IERC721(collection).safeTransferNFT(address(this), to, tokenId);
    }

    // 编辑订单的ETH（修改Bid订单价格时调用）
    function editETH(
        OrderKey oldOrderKey, // 旧订单标识
        OrderKey newOrderKey, // 新订单标识
        uint256 oldETHAmount, // 旧订单的ETH金额
        uint256 newETHAmount, // 新订单的ETH金额
        address to            // 接收退款的地址
    ) external payable onlyEasySwapOrderBook {
        // 清零旧订单的ETH余额
        ETHBalance[oldOrderKey] = 0;
        
        // 情况1：新价格更低（需要退款）
        if (oldETHAmount > newETHAmount) {
            // 设置新订单余额为新金额
            ETHBalance[newOrderKey] = newETHAmount;
            // 退款差额给用户
            to.safeTransferETH(oldETHAmount - newETHAmount);
        } 
        // 情况2：新价格更高（需要补足）
        else if (oldETHAmount < newETHAmount) {
            // 验证发送的ETH足够补足差额
            require(
                msg.value >= newETHAmount - oldETHAmount,
                "HV: not match newETHAmount"
            );
            // 设置新订单余额为旧余额加上补充金额
            ETHBalance[newOrderKey] = msg.value + oldETHAmount;
        } 
        // 情况3：价格不变
        else {
            // 直接转移余额到新订单
            ETHBalance[newOrderKey] = oldETHAmount;
        }
    }

    // 编辑订单的NFT（修改List订单时调用）
    function editNFT(
        OrderKey oldOrderKey, // 旧订单标识
        OrderKey newOrderKey  // 新订单标识
    ) external onlyEasySwapOrderBook {
        // 将NFT归属从旧订单转移到新订单
        NFTBalance[newOrderKey] = NFTBalance[oldOrderKey];
        // 清除旧订单的NFT记录
        delete NFTBalance[oldOrderKey];
    }

    // 直接转移NFT（用于即时交易，不经过存提流程）
    function transferERC721(
        address from,                  // NFT发送者
        address to,                    // NFT接收者
        LibOrder.Asset calldata assets // NFT资产信息
    ) external onlyEasySwapOrderBook {
        // 安全地将NFT直接从发送者转移到接收者
        IERC721(assets.collection).safeTransferNFT(from, to, assets.tokenId);
    }

    // 批量转移NFT（用于批量交易）
    function batchTransferERC721(
        address to,                         // NFT接收者
        LibOrder.NFTInfo[] calldata assets  // NFT信息数组
    ) external {
        // 循环处理每个NFT
        for (uint256 i = 0; i < assets.length; ++i) {
            // 安全转移NFT（注意：这里从调用者转移，而不是从金库！）
            IERC721(assets[i].collection).safeTransferNFT(
                _msgSender(),           // ❗问题：从调用者转移，应指定明确来源
                to,                     // 接收者
                assets[i].tokenId       // NFT tokenId
            );
        }
    }

    // ERC721接收回调函数（必须实现以安全接收NFT）
    function onERC721Received(
        address,     // NFT发送者（忽略）
        address,     // 发送前的所有者（忽略）
        uint256,     // tokenId（忽略）
        bytes memory // 附加数据（忽略）
    ) public virtual returns (bytes4) {
        // 返回函数选择器，表示成功接收NFT
        return this.onERC721Received.selector;
    }

    // 接收ETH的回退函数（允许合约接收ETH）
    receive() external payable {}

    // 可升级合约的存储间隙（预留50个存储槽位供未来升级使用）
    uint256[50] private __gap;
}

**3）OrderStorage  合约**：
// SPDX-License-Identifier: MIT
// MIT许可证标识
pragma solidity ^0.8.19;
// 指定Solidity编译器版本为0.8.19及以上

// 导入OpenZeppelin可升级合约的初始化器
import {Initializable} from "@openzeppelin/contracts-upgradeable/proxy/utils/Initializable.sol";

// 导入自定义库：红黑树库（用于价格排序）和订单库
import {RedBlackTreeLibrary, Price} from "./libraries/RedBlackTreeLibrary.sol";
import {LibOrder, OrderKey} from "./libraries/LibOrder.sol";

// 自定义错误：不能插入重复的订单
error CannotInsertDuplicateOrder(OrderKey orderKey);

// OrderStorage合约，继承Initializable以实现可升级
contract OrderStorage is Initializable {
    // 使用RedBlackTreeLibrary扩展Tree类型的功能
    using RedBlackTreeLibrary for RedBlackTreeLibrary.Tree;

    /// @dev 所有订单键都包装在一个哨兵值中以避免冲突
    // 映射：订单键 → 订单详情（包含订单数据和指向下一个订单的指针）
    mapping(OrderKey => LibOrder.DBOrder) public orders;

    /// @dev 每个NFT集合和订单方向的价格树，按价格排序
    // 双重映射：NFT集合地址 → 订单方向 → 价格红黑树
    mapping(address => mapping(LibOrder.Side => RedBlackTreeLibrary.Tree))
        public priceTrees;

    /// @dev 每个NFT集合、订单方向和特定价格的订单队列，按订单键排序
    // 三重映射：NFT集合 → 订单方向 → 价格 → 订单队列
    mapping(address => mapping(LibOrder.Side => mapping(Price => LibOrder.OrderQueue)))
        public orderQueues;

    // 初始化函数（可升级合约模式）
    function __OrderStorage_init() internal onlyInitializing {}

    // 非链式初始化函数
    function __OrderStorage_init_unchained() internal onlyInitializing {}

    // 内部工具函数：安全的加1操作（使用unchecked避免溢出检查节省gas）
    function onePlus(uint256 x) internal pure returns (uint256) {
        unchecked {
            return 1 + x;
        }
    }

    // 获取最佳价格（对于买单是最高价，对于卖单是最低价）
    function getBestPrice(
        address collection, // NFT集合地址
        LibOrder.Side side  // 订单方向：Bid(买单)或List(卖单)
    ) public view returns (Price price) {
        // 如果是买单：返回价格树中的最后一个（最高价）
        // 如果是卖单：返回价格树中的第一个（最低价）
        price = (side == LibOrder.Side.Bid)
            ? priceTrees[collection][side].last()  // 买单：最高价
            : priceTrees[collection][side].first(); // 卖单：最低价
    }

    // 获取下一个最佳价格
    function getNextBestPrice(
        address collection,  // NFT集合地址
        LibOrder.Side side,  // 订单方向
        Price price          // 当前价格
    ) public view returns (Price nextBestPrice) {
        // 如果当前价格为空
        if (RedBlackTreeLibrary.isEmpty(price)) {
            // 返回最佳价格（类似getBestPrice）
            nextBestPrice = (side == LibOrder.Side.Bid)
                ? priceTrees[collection][side].last()
                : priceTrees[collection][side].first();
        } else {
            // 返回当前价格的下一个价格
            // 买单：更低价（递减）
            // 卖单：更高价（递增）
            nextBestPrice = (side == LibOrder.Side.Bid)
                ? priceTrees[collection][side].prev(price)  // 买单：前一个价格（更低价）
                : priceTrees[collection][side].next(price); // 卖单：后一个价格（更高价）
        }
    }

    // 内部函数：添加订单到存储系统
    function _addOrder(
        LibOrder.Order memory order  // 要添加的订单
    ) internal returns (OrderKey orderKey) {
        // 获取订单的哈希值作为唯一键
        orderKey = LibOrder.hash(order);
        
        // 检查订单是否已经存在（通过maker地址是否为0判断）
        if (orders[orderKey].order.maker != address(0)) {
            // 如果已存在，抛出错误
            revert CannotInsertDuplicateOrder(orderKey);
        }

        // 如果价格树中不存在这个价格，则插入价格
        RedBlackTreeLibrary.Tree storage priceTree = priceTrees[
            order.nft.collection  // NFT集合地址
        ][order.side];  // 订单方向
        
        if (!priceTree.exists(order.price)) {
            priceTree.insert(order.price);  // 将价格插入红黑树
        }

        // 将订单插入订单队列
        LibOrder.OrderQueue storage orderQueue = orderQueues[
            order.nft.collection  // NFT集合
        ][order.side]  // 订单方向
        [order.price];  // 价格

        // 如果队列未初始化（头尾都是哨兵）
        if (LibOrder.isSentinel(orderQueue.head)) {
            // 创建新的队列（头尾都设为哨兵）
            orderQueues[order.nft.collection][order.side][
                order.price
            ] = LibOrder.OrderQueue(
                LibOrder.ORDERKEY_SENTINEL,  // 头：哨兵
                LibOrder.ORDERKEY_SENTINEL   // 尾：哨兵
            );
            // 重新获取队列引用
            orderQueue = orderQueues[order.nft.collection][order.side][
                order.price
            ];
        }
        
        // 如果队列为空（尾是哨兵）
        if (LibOrder.isSentinel(orderQueue.tail)) {
            // 设置队列头尾都为新订单键
            orderQueue.head = orderKey;
            orderQueue.tail = orderKey;
            
            // 存储订单数据，下一个订单指向哨兵
            orders[orderKey] = LibOrder.DBOrder(
                order,  // 订单数据
                LibOrder.ORDERKEY_SENTINEL  // 下一个订单：哨兵
            );
        } else {
            // 队列不为空：将新订单插入队列尾部
            // 更新原尾部订单的next指针
            orders[orderQueue.tail].next = orderKey;
            
            // 存储新订单数据，下一个订单指向哨兵
            orders[orderKey] = LibOrder.DBOrder(
                order,
                LibOrder.ORDERKEY_SENTINEL
            );
            
            // 更新队列尾部为新订单
            orderQueue.tail = orderKey;
        }
    }

    // 内部函数：从存储系统中移除订单
    function _removeOrder(
        LibOrder.Order memory order  // 要移除的订单
    ) internal returns (OrderKey orderKey) {
        // 获取对应的订单队列
        LibOrder.OrderQueue storage orderQueue = orderQueues[
            order.nft.collection
        ][order.side][order.price];
        
        // 从队列头部开始遍历
        orderKey = orderQueue.head;
        OrderKey prevOrderKey;  // 前一个订单键
        bool found;  // 是否找到目标订单

        // 遍历队列查找目标订单
        while (LibOrder.isNotSentinel(orderKey) && !found) {
            LibOrder.DBOrder memory dbOrder = orders[orderKey];
            
            // 比较订单的所有属性是否匹配
            if (
                (dbOrder.order.maker == order.maker) &&
                (dbOrder.order.saleKind == order.saleKind) &&
                (dbOrder.order.expiry == order.expiry) &&
                (dbOrder.order.salt == order.salt) &&
                (dbOrder.order.nft.tokenId == order.nft.tokenId) &&
                (dbOrder.order.nft.amount == order.nft.amount)
            ) {
                // 找到目标订单
                OrderKey temp = orderKey;  // 临时保存当前订单键
                
                // 如果目标订单是队列头部
                if (
                    OrderKey.unwrap(orderQueue.head) ==
                    OrderKey.unwrap(orderKey)
                ) {
                    orderQueue.head = dbOrder.next;  // 更新队列头部
                } else {
                    // 更新前一个订单的next指针，跳过当前订单
                    orders[prevOrderKey].next = dbOrder.next;
                }
                
                // 如果目标订单是队列尾部
                if (
                    OrderKey.unwrap(orderQueue.tail) ==
                    OrderKey.unwrap(orderKey)
                ) {
                    orderQueue.tail = prevOrderKey;  // 更新队列尾部
                }
                
                // 保存当前订单键到prevOrderKey
                prevOrderKey = orderKey;
                // 移动到下一个订单
                orderKey = dbOrder.next;
                // 删除订单数据
                delete orders[temp];
                found = true;  // 标记已找到
            } else {
                // 未匹配，继续遍历
                prevOrderKey = orderKey;
                orderKey = dbOrder.next;
            }
        }
        
        // 如果找到了订单
        if (found) {
            // 如果队列现在为空（头部是哨兵）
            if (LibOrder.isSentinel(orderQueue.head)) {
                // 删除空队列
                delete orderQueues[order.nft.collection][order.side][
                    order.price
                ];
                
                // 从价格树中移除该价格（如果存在）
                RedBlackTreeLibrary.Tree storage priceTree = priceTrees[
                    order.nft.collection
                ][order.side];
                if (priceTree.exists(order.price)) {
                    priceTree.remove(order.price);
                }
            }
        } else {
            // 未找到订单，抛出错误
            revert("Cannot remove missing order");
        }
    }

    /**
     * @dev 检索符合指定条件的订单列表
     * @param collection NFT集合地址
     * @param tokenId NFT的ID
     * @param side 要检索的订单方向（买单或卖单）
     * @param saleKind 销售类型（固定价格或拍卖）
     * @param count 要检索的最大订单数
     * @param price 要检索的订单的最高价格
     * @param firstOrderKey 要检索的第一个订单的键
     * @return resultOrders 符合指定条件的订单数组
     * @return nextOrderKey 下一个要检索的订单的键（用于分页）
     */
    function getOrders(
        address collection,          // NFT集合地址
        uint256 tokenId,             // NFT的tokenId
        LibOrder.Side side,          // 订单方向
        LibOrder.SaleKind saleKind,  // 销售类型
        uint256 count,               // 最大返回数量
        Price price,                 // 价格筛选条件
        OrderKey firstOrderKey       // 起始订单键（分页用）
    )
        external
        view
        returns (LibOrder.Order[] memory resultOrders, OrderKey nextOrderKey)
    {
        // 初始化结果数组
        resultOrders = new LibOrder.Order[](count);

        // 如果price参数为空，获取最佳价格
        if (RedBlackTreeLibrary.isEmpty(price)) {
            price = getBestPrice(collection, side);
        } else {
            // 如果指定了firstOrderKey且price不为空，获取下一个最佳价格
            if (LibOrder.isSentinel(firstOrderKey)) {
                price = getNextBestPrice(collection, side, price);
            }
        }

        uint256 i;  // 结果数组索引
        // 遍历价格树，直到填满结果数组或没有更多价格
        while (RedBlackTreeLibrary.isNotEmpty(price) && i < count) {
            // 获取当前价格的订单队列
            LibOrder.OrderQueue memory orderQueue = orderQueues[collection][
                side
            ][price];
            
            OrderKey orderKey = orderQueue.head;  // 从队列头部开始
            
            // 如果指定了起始订单键，需要找到该订单在队列中的位置
            if (LibOrder.isNotSentinel(firstOrderKey)) {
                while (
                    LibOrder.isNotSentinel(orderKey) &&
                    OrderKey.unwrap(orderKey) != OrderKey.unwrap(firstOrderKey)
                ) {
                    LibOrder.DBOrder memory order = orders[orderKey];
                    orderKey = order.next;  // 移动到下一个订单
                }
                firstOrderKey = LibOrder.ORDERKEY_SENTINEL;  // 重置起始键
            }

            // 遍历当前价格的所有订单
            while (LibOrder.isNotSentinel(orderKey) && i < count) {
                LibOrder.DBOrder memory dbOrder = orders[orderKey];
                orderKey = dbOrder.next;  // 提前移动到下一个订单
                
                // 跳过已过期的订单（如果有过期时间且已过期）
                if (
                    (dbOrder.order.expiry != 0 &&
                        dbOrder.order.expiry < block.timestamp)
                ) {
                    continue;  // 跳过这个订单
                }

                // 特殊处理：如果是买单且saleKind是FixedPriceForCollection
                // 跳过类型为FixedPriceForItem的订单
                if (
                    (side == LibOrder.Side.Bid) &&
                    (saleKind == LibOrder.SaleKind.FixedPriceForCollection)
                ) {
                    if (
                        (dbOrder.order.side == LibOrder.Side.Bid) &&
                        (dbOrder.order.saleKind ==
                            LibOrder.SaleKind.FixedPriceForItem)
                    ) {
                        continue;  // 跳过这个订单
                    }
                }

                // 特殊处理：如果是买单且saleKind是FixedPriceForItem
                // 跳过tokenId不匹配的订单
                if (
                    (side == LibOrder.Side.Bid) &&
                    (saleKind == LibOrder.SaleKind.FixedPriceForItem)
                ) {
                    if (
                        (dbOrder.order.side == LibOrder.Side.Bid) &&
                        (dbOrder.order.saleKind ==
                            LibOrder.SaleKind.FixedPriceForItem) &&
                        (tokenId != dbOrder.order.nft.tokenId)
                    ) {
                        continue;  // 跳过这个订单
                    }
                }

                // 将订单添加到结果数组
                resultOrders[i] = dbOrder.order;
                // 记录下一个订单键（用于分页）
                nextOrderKey = dbOrder.next;
                // 递增索引（使用安全加法）
                i = onePlus(i);
            }
            
            // 移动到下一个价格
            price = getNextBestPrice(collection, side, price);
        }
    }

    // 获取最佳订单（价格最优且未过期的订单）
    function getBestOrder(
        address collection,          // NFT集合地址
        uint256 tokenId,             // NFT的tokenId
        LibOrder.Side side,          // 订单方向
        LibOrder.SaleKind saleKind   // 销售类型
    ) external view returns (LibOrder.Order memory orderResult) {
        // 获取最佳价格
        Price price = getBestPrice(collection, side);
        
        // 遍历价格树
        while (RedBlackTreeLibrary.isNotEmpty(price)) {
            // 获取当前价格的订单队列
            LibOrder.OrderQueue memory orderQueue = orderQueues[collection][
                side
            ][price];
            
            OrderKey orderKey = orderQueue.head;  // 从队列头部开始
            
            // 遍历当前价格的所有订单
            while (LibOrder.isNotSentinel(orderKey)) {
                LibOrder.DBOrder memory dbOrder = orders[orderKey];
                
                // 特殊处理：如果是买单且saleKind是FixedPriceForItem
                // 跳过tokenId不匹配的订单
                if (
                    (side == LibOrder.Side.Bid) &&
                    (saleKind == LibOrder.SaleKind.FixedPriceForItem)
                ) {
                    if (
                        (dbOrder.order.side == LibOrder.Side.Bid) &&
                        (dbOrder.order.saleKind ==
                            LibOrder.SaleKind.FixedPriceForItem) &&
                        (tokenId != dbOrder.order.nft.tokenId)
                    ) {
                        orderKey = dbOrder.next;  // 跳过，继续下一个
                        continue;
                    }
                }

                // 特殊处理：如果是买单且saleKind是FixedPriceForCollection
                // 跳过类型为FixedPriceForItem的订单
                if (
                    (side == LibOrder.Side.Bid) &&
                    (saleKind == LibOrder.SaleKind.FixedPriceForCollection)
                ) {
                    if (
                        (dbOrder.order.side == LibOrder.Side.Bid) &&
                        (dbOrder.order.saleKind ==
                            LibOrder.SaleKind.FixedPriceForItem)
                    ) {
                        orderKey = dbOrder.next;  // 跳过，继续下一个
                        continue;
                    }
                }

                // 检查订单是否未过期（或永不过期）
                if (
                    (dbOrder.order.expiry == 0 ||
                        dbOrder.order.expiry > block.timestamp)
                ) {
                    orderResult = dbOrder.order;  // 找到符合条件的订单
                    break;  // 跳出内层循环
                }
                
                orderKey = dbOrder.next;  // 继续下一个订单
            }
            
            // 如果找到了订单（价格>0），跳出外层循环
            if (Price.unwrap(orderResult.price) > 0) {
                break;
            }
            
            // 移动到下一个价格
            price = getNextBestPrice(collection, side, price);
        }
    }

    // 存储间隙：预留50个存储槽供未来升级使用
    uint256[50] private __gap;
}

  三层索引结构
1. 价格树 (priceTrees)
   - 第一层: NFT集合地址
   - 第二层: 订单方向 (Bid/List)
   - 第三层: 红黑树，按价格排序
   
2. 订单队列 (orderQueues)
   - 第一层: NFT集合地址
   - 第二层: 订单方向
   - 第三层: 价格
   - 第四层: 订单链表（先进先出）

3. 订单映射 (orders)
   - 键: OrderKey（订单哈希）
   - 值: DBOrder（订单数据 + 下一个订单指针）
排序规则
买单 (Bid): 价格从高到低排序（出价高的优先）

卖单 (List): 价格从低到高排序（价格低的优先）

哨兵值设计

避免与有效的订单键冲突

/**
 * @title 订单参数有效性验证抽象合约
 * @dev 验证订单参数的合法性，跟踪订单的成交状态和取消状态
 */
abstract contract OrderValidator is
    Initializable,           // 可升级合约初始化
    ContextUpgradeable,     // 上下文管理
    EIP712Upgradeable       // EIP712签名支持
{
    // EIP-1271魔法值，用于合约签名验证（标准合约签名返回值）
    bytes4 private constant EIP_1271_MAGIC_VALUE = 0x1626ba7e;

    // 已取消订单的标记值（使用uint256最大值表示已取消）
    uint256 private constant CANCELLED = type(uint256).max;

    // 订单成交状态记录映射
    // 键：订单哈希(OrderKey)
    // 值：已成交数量（如果值为CANCELLED，表示订单已被取消）
    mapping(OrderKey => uint256) public filledAmount;

    /**
     * @notice 订单验证器初始化函数
     * @param EIP712Name EIP712域名名称
     * @param EIP712Version EIP712版本号
     */
    function __OrderValidator_init(
        string memory EIP712Name,
        string memory EIP712Version
    ) internal onlyInitializing {
        // 初始化上下文
        __Context_init();
        // 初始化EIP712，设置域名和版本
        __EIP712_init(EIP712Name, EIP712Version);
        // 执行非链式初始化
        __OrderValidator_init_unchained();
    }

    /**
     * @notice 订单验证器非链式初始化函数
     */
    function __OrderValidator_init_unchained() internal onlyInitializing {}

    /**
     * @notice 验证订单参数的有效性
     * @param order 要验证的订单
     * @param isSkipExpiry 是否跳过过期时间检查
     */
    function _validateOrder(
        LibOrder.Order memory order,
        bool isSkipExpiry
    ) internal view {
        // 订单必须包含创建者地址
        require(order.maker != address(0), "OVa: miss maker");

        // 如果不是跳过过期检查，则需要验证过期时间
        if (!isSkipExpiry) {
            require(
                order.expiry == 0 || order.expiry > block.timestamp,
                "OVa: expired"
            );
            // expiry为0表示永不过期
            // 非0值必须大于当前区块时间戳
        }

        // 订单的随机盐值不能为0（防止重放攻击）
        require(order.salt != 0, "OVa: zero salt");

        // 根据订单类型进行特定验证
        if (order.side == LibOrder.Side.List) {
            // 挂单（卖单）：必须指定有效的NFT集合地址
            require(
                order.nft.collection != address(0),
                "OVa: unsupported nft asset"
            );
        } else if (order.side == LibOrder.Side.Bid) {
            // 出价（买单）：价格必须大于0
            require(Price.unwrap(order.price) > 0, "OVa: zero price");
        }
    }

    /**
     * @notice 获取订单的已成交数量
     * @param orderKey 订单哈希
     * @return orderFilledAmount 已成交数量（如果订单未成交返回0）
     */
    function _getFilledAmount(
        OrderKey orderKey
    ) internal view returns (uint256 orderFilledAmount) {
        // 从映射中读取已成交数量
        orderFilledAmount = filledAmount[orderKey];
        // 已取消的订单不能被匹配
        require(orderFilledAmount != CANCELLED, "OVa: canceled");
    }

    /**
     * @notice 更新订单的已成交数量
     * @param newAmount 新的成交数量
     * @param orderKey 订单哈希
     */
    function _updateFilledAmount(
        uint256 newAmount,
        OrderKey orderKey
    ) internal {
        // 防止将订单状态设置为已取消（应该通过_cancelOrder专门函数）
        require(newAmount != CANCELLED, "OVa: canceled");
        // 更新成交数量
        filledAmount[orderKey] = newAmount;
    }

    /**
     * @notice 取消订单
     * @dev 已取消的订单不能被重新激活
     * @param orderKey 订单哈希
     */
    function _cancelOrder(OrderKey orderKey) internal {
        // 将订单状态标记为已取消
        filledAmount[orderKey] = CANCELLED;
    }

    // 存储间隙：预留50个存储槽供未来升级使用
    uint256[50] private __gap;
}

        );

    function __ProtocolManager_init_unchained(
