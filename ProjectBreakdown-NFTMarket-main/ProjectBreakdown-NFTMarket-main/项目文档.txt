// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;
import {OrderKey, Price, LibOrder} from "../libraries/LibOrder.sol";
// 导入订单库，包含：
// - OrderKey: 订单唯一标识符类型
// - Price: 价格类型（可能是 uint256 的包装类型）
// - LibOrder: 订单库，包含订单结构体和相关函数

interface IEasySwapOrderBook {
    // 定义订单簿接口，这是去中心化交易所的核心合约
    
    function makeOrders(
        LibOrder.Order[] calldata newOrders
    ) external payable returns (OrderKey[] memory newOrderKeys);
    // 创建多个订单
    // external: 只能从外部调用
    // payable: 可以接收 ETH（用于 Bid 订单）
    // 参数: newOrders - 订单数组，使用 calldata 节省 gas
    // 返回: 新创建的订单键数组
    // 功能: 创建订单并转移相关资产（NFT 或 ETH）

    function cancelOrders(
        OrderKey[] calldata orderKeys
    ) external returns (bool[] memory successes);
    // 取消多个订单
    // 只有订单创建者可以取消自己的订单
    // 返回每个订单是否成功取消的布尔数组

    function editOrders(
        LibOrder.EditDetail[] calldata editDetails
    ) external payable returns (OrderKey[] memory newOrderKeys);
    // 编辑多个订单
    // 实际上是取消旧订单 + 创建新订单的组合操作
    // payable: 如果新订单价格更高（Bid）或需要补充 ETH，需要发送 ETH
    // 限制: 只能修改价格，其他属性必须保持不变

    function matchOrder(
        LibOrder.Order calldata sellOrder,
        LibOrder.Order calldata buyOrder
    ) external payable;
    // 匹配单个订单对（简化版）
    // 将卖单和买单进行匹配，完成交易
    // payable: 买家需要发送 ETH 来购买

    function matchOrders(
        LibOrder.MatchDetail[] calldata matchDetails
    ) external payable returns (bool[] memory successes);
    // 批量匹配多个订单对
    // 原子性操作：要么全部成功，要么全部失败
    // 支持批量交易，提高效率
}

// LibOrder.Side 枚举
enum Side {
    List,  // 挂单：卖方列出 NFT 出售
    Bid    // 出价：买方出价购买 NFT
}

//  LibOrder.SaleKind 枚举
enum SaleKind {
    FixedPriceForItem,     // 固定价格
    FixedPriceForCollection, // 集合定价
    DutchAuction,          // 荷兰拍卖
    EnglishAuction         // 英式拍卖
}
